61 {

76  run_results_header_to_data->at(std::string(run_eval::kRunInputSigHeaders[2]))[num_run]});
77

109 //get mapping of headers to data in csv file for run results and speedups

112 std::map<std::string, std::vector<std::string>>
113 RunResultsSpeedups::HeaderToDataInCsvFile(
114  const std::filesystem::path& csv_file_path) const
115 {
116  std::ifstream csv_file_str(csv_file_path);
117  if (!(csv_file_str.is_open())) {
118  std::cout << "ERROR CREATING STREAM: " << csv_file_path << std::endl;
119  }
120  //retrieve data headers from top row
121  std::string headers_line;
122  std::getline(csv_file_str, headers_line);
123  std::stringstream headers_str(headers_line);
124  std::vector<std::string> data_headers;
125  std::string header;
126  std::map<std::string, std::vector<std::string>> header_to_data;
127  while (std::getline(headers_str, header, ',')) {
128  data_headers.push_back(header);
129  header_to_data[header] = std::vector<std::string>();
130  }
131  //go through each data line and add to mapping from headers to data
132  std::string data_line;
133  while (std::getline(csv_file_str, data_line)) {
134  std::stringstream data_line_str(data_line);
135  std::string data;
136  unsigned int num_data{0};
137  while (std::getline(data_line_str, data, ',')) {
138  header_to_data[data_headers[num_data++]].push_back(data);
139  }
140  }
141  return header_to_data;
142 }
