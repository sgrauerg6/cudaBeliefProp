168 //get bp level properties for next (higher) level in hierarchy that processes data with half width/height of current level
169 template <RunData_t T>
170 BpLevel<T> BpLevel<T>::NextBpLevel(unsigned int num_disparity_values) const
171 {
172  const std::size_t offset_next_level =
173  level_properties_.offset_into_arrays_ +
174  NumDataInBpArrays(num_disparity_values);
175  return BpLevel<T>(
176  {(unsigned int)std::ceil((float)level_properties_.width_level_ / 2.0f),
177  (unsigned int)std::ceil((float)level_properties_.height_level_ / 2.0f)},
178  offset_next_level,
179  (level_properties_.level_num_ + 1),
180  level_properties_.bytes_align_memory_);

194 template <RunData_t T>
195 std::size_t BpLevel<T>::NumDataForAlignedMemoryAtLevel(
196  const std::array<unsigned int, 2>& width_height_level,
197  unsigned int num_possible_disparities) const
198 {
199  const std::size_t numDataAtLevel =
200  (std::size_t)PaddedCheckerboardWidth(CheckerboardWidthTargetDevice(width_height_level[0])) *
201  ((std::size_t)width_height_level[1]) *
202  (std::size_t)num_possible_disparities;
203  std::size_t numBytesAtLevel = numDataAtLevel * sizeof(T);
204 
205  if ((numBytesAtLevel % level_properties_.bytes_align_memory_) == 0) {
206  return numDataAtLevel;
207  }
208  else {
209  numBytesAtLevel +=
210  (level_properties_.bytes_align_memory_ -
211  (numBytesAtLevel % level_properties_.bytes_align_memory_));
212  return (numBytesAtLevel / sizeof(T));
213  }
214 }

223 template <RunData_t T>
224 unsigned int BpLevel<T>::PaddedCheckerboardWidth(
225  unsigned int checkerboard_width) const
226 {
227  size_t num_data_align_width = level_properties_.bytes_align_memory_ / sizeof(T);
228  //add "padding" to checkerboard width if necessary for alignment
229  return ((checkerboard_width % num_data_align_width) == 0) ?
230  checkerboard_width :
231  (checkerboard_width +
232  (num_data_align_width - (checkerboard_width % num_data_align_width)));
233 }

235 //static function
236 template <RunData_t T>
237 std::size_t BpLevel<T>::TotalDataForAlignedMemoryAllLevels(
238  const std::array<unsigned int, 2>& width_height_bottom_level,
239  unsigned int num_possible_disparities,
240  unsigned int num_levels,
241  run_environment::AccSetting acceleration)
242 {
243  BpLevel<T> curr_level_properties(width_height_bottom_level, 0, 0, acceleration);
244  std::size_t total_data =
245  curr_level_properties.NumDataInBpArrays(num_possible_disparities);
246  for (unsigned int curr_level = 1; curr_level < num_levels; curr_level++)
247  {
248  curr_level_properties = curr_level_properties.NextBpLevel(num_possible_disparities);
249  total_data += curr_level_properties.NumDataInBpArrays(num_possible_disparities);
250  }
251 
252  return total_data;
253 }
