161 template<> inline void beliefprop_cpu::UpdateBestDispBestVals<float32x4_t>(float32x4_t& best_disparities, float32x4_t& best_vals,
162  const float32x4_t& current_disparity, const float32x4_t& val_at_disp)
163 {
164  //get mask with value 1 where current value less then current best 1, 0 otherwise
165  uint32x4_t mask_update_vals = vcltq_f32(val_at_disp, best_vals);
166  //update best values and best disparities using mask
167  //vbslq_f32 operation uses first float32x4_t argument if mask value is 1 and seconde float32x4_t argument if mask value is 0
168  best_vals = vbslq_f32(mask_update_vals, val_at_disp, best_vals);
169  best_disparities = vbslq_f32(mask_update_vals, current_disparity, best_disparities);
170 }
