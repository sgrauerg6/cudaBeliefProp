35 
36  //initialize total disparity difference across all pixels with and without max for disparity difference to 0
37  std::array<float, 2> total_disp_abs_diff_no_max_w_max{0, 0};
38 
39  //go through each disparity map output pixel and evaluate output
40  //against corresponding output in disparity map to compare
41  for (unsigned int i = 0; i < this->TotalPixels(); i++) {
42  //get disparity difference between disparity maps at pixel i
43  const T disp_map_val = (this->pixels_.get())[i];
44  const T disp_map_compare_val = disparity_map_to_compare.PixelAtPoint(i);
45  const T abs_diff = std::abs(disp_map_val - disp_map_compare_val);
46 
47  //add disparity difference at pixel to total disparity difference across pixels
48  //with and without max disparity difference
49  total_disp_abs_diff_no_max_w_max[0] += abs_diff;
50  total_disp_abs_diff_no_max_w_max[1] += std::min(abs_diff, evaluation_parameters.max_diff_cap);
51 
52  //increment number of pixels that differ greater than threshold for each threshold
53  std::ranges::for_each(evaluation_parameters.output_diff_thresholds,
54  [abs_diff, &output_evaluation](auto threshold) {
55  if (abs_diff > threshold) {
56  output_evaluation.num_sig_diff_pixels_at_thresholds_[threshold]++;
57  }
58  });
59  }
60 
61  output_evaluation.average_disp_abs_diff_no_max_w_max_[0] = total_disp_abs_diff_no_max_w_max[0] / this->TotalPixels();
62  output_evaluation.average_disp_abs_diff_no_max_w_max_[1] = total_disp_abs_diff_no_max_w_max[1] / this->TotalPixels();
63 
64  //need to cast unsigned ints to float to get proportion of pixels that differ by more than threshold
65  std::ranges::transform(output_evaluation.num_sig_diff_pixels_at_thresholds_,
66  std::inserter(output_evaluation.prop_sig_diff_pixels_at_thresholds_, output_evaluation.prop_sig_diff_pixels_at_thresholds_.end()),
67  [this](const auto& sig_diff_pixel_at_threshold_map) -> std::pair<float, float> {
68  return { sig_diff_pixel_at_threshold_map.first, ((float)sig_diff_pixel_at_threshold_map.second) / ((float)(this->TotalPixels())) };
69  });
70 
71  return output_evaluation;
72 }
