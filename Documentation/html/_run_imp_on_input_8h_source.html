113  MultRunData::mapped_type RunEvalBenchmark(
114  const run_environment::RunImpSettings& run_imp_settings,
115  bool run_w_loop_iters_templated) const
116  {
117  MultRunData::mapped_type::value_type out_run_data;
118  enum class RunType { ONLY_RUN, DEFAULT_PARAMS, OPTIMIZED_RUN, TEST_PARAMS };
119 
120  //set up parallel parameters for specific benchmark
121  std::shared_ptr<ParallelParams> parallel_params = SetUpParallelParams(run_imp_settings);
122 
123  //if optimizing parallel parameters, parallel_params_vect contains parallel parameter settings to run
124  //(and contains only the default parallel parameters if not)
125  std::vector<std::array<unsigned int, 2>> parallel_params_vect{
126  run_imp_settings.opt_parallel_params_setting.first ?
127  run_imp_settings.p_params_default_opt_settings.second :
128  std::vector<std::array<unsigned int, 2>>()};
129 
130  //if optimizing parallel parameters, run BP for each parallel parameters option,
131  //retrieve best parameters for each kernel or overall for the run,
132  //and then run BP with best found parallel parameters
133  //if not optimizing parallel parameters, run BP once using default parallel parameters
134  for (unsigned int run_num = 0; run_num < (parallel_params_vect.size() + 1); run_num++) {
135  //initialize current run type to specify if current run is only run, run with
136  //default params, test params run, or final run with optimized params
137  RunType curr_run_type{RunType::TEST_PARAMS};
138  if (!run_imp_settings.opt_parallel_params_setting.first) {
139  curr_run_type = RunType::ONLY_RUN;
140  }
141  else if (run_num == parallel_params_vect.size()) {
142  curr_run_type = RunType::OPTIMIZED_RUN;
143  }
144 
145  //get and set parallel parameters for current run if not final run that uses optimized parameters
146  std::array<unsigned int, 2> p_params_curr_run{run_imp_settings.p_params_default_opt_settings.first};
147  if (curr_run_type == RunType::ONLY_RUN) {
148  parallel_params->SetParallelDims(run_imp_settings.p_params_default_opt_settings.first);
149  }
150  else if (curr_run_type == RunType::TEST_PARAMS) {
151  //set parallel parameters to parameters corresponding to current run for each BP processing level
152  p_params_curr_run = parallel_params_vect[run_num];
153  parallel_params->SetParallelDims(p_params_curr_run);
154  if (p_params_curr_run == run_imp_settings.p_params_default_opt_settings.first) {
155  //set run type to default parameters if current run uses default parameters
156  curr_run_type = RunType::DEFAULT_PARAMS;
157  }
158  }
159 
160  //store input params data if using default parallel parameters or final run with optimized parameters
161  RunData curr_run_data;
162  if (curr_run_type != RunType::TEST_PARAMS) {
163  //add input and parameters data for specific benchmark to current run data
164  curr_run_data.AddDataWHeader(
165  std::string(run_eval::kInputIdxHeader),
166  std::to_string(NUM_INPUT));
167  curr_run_data.AppendData(InputAndParamsForCurrBenchmark(run_w_loop_iters_templated));
168  if ((run_imp_settings.opt_parallel_params_setting.first) &&
169  (run_imp_settings.opt_parallel_params_setting.second ==
170  run_environment::OptParallelParamsSetting::kAllowDiffKernelParallelParamsInRun))
171  {
172  //add parallel parameters for each kernel to current input data if
173  //allowing different parallel parameters for each kernel
174  //in the same run
175  curr_run_data.AppendData(parallel_params->AsRunData());
176  }
177  }
178 
179  //run only optimized implementation and not single-threaded run if
180  //current run is not final run or is using default parameter parameters
181  const bool run_opt_imp_only{curr_run_type == RunType::TEST_PARAMS};
182 
183  //run benchmark implementation(s) and return null output if error in run
184  //detailed results stored to file that is generated using stream
185  const auto run_imps_results =
186  RunImpsAndCompare(parallel_params, run_opt_imp_only, run_w_loop_iters_templated);
187  curr_run_data.AddDataWHeader(
188  std::string(run_eval::kRunSuccessHeader),
189  run_imps_results.has_value());
190 
191  //if error in run and run is any type other than for testing parameters,
192  //exit function with null output to indicate error
193  if ((!run_imps_results) && (curr_run_type != RunType::TEST_PARAMS)) {
194  return {};
195  }
196 
197  //add data results from current run if run successful
198  if (run_imps_results) {
199  curr_run_data.AppendData(run_imps_results.value());
200  }
201 
202  //add current run results for output if using default parallel
203  //parameters or is final run w/ optimized parallel parameters
204  if (curr_run_type != RunType::TEST_PARAMS) {
205  //set output for runs using default parallel parameters and final run
206  //(which is the same run if not optimizing parallel parameters)
207  if (curr_run_type == RunType::OPTIMIZED_RUN) {
208  out_run_data[run_environment::ParallelParamsSetting::kOptimized] = curr_run_data;
209  }
210  else {
211  out_run_data[run_environment::ParallelParamsSetting::kDefault] = curr_run_data;
212  if (curr_run_type == RunType::ONLY_RUN) {
213  //if current run is only run due to setting of not optimizing parallel parameters,
214  //set optimized parallel parameters output to current run data, resulting in same
215  //run data mapped to default and optimized parallel parameters enums
216  out_run_data[run_environment::ParallelParamsSetting::kOptimized] = curr_run_data;
217  }
218  }
219  }
220 
221  if (run_imp_settings.opt_parallel_params_setting.first) {
222  //retrieve and store results including runtimes for each kernel if
223  //allowing different parallel parameters for each kernel and
224  //total runtime for current run
225  //if error in run, don't add results for current parallel parameters to results set
226  if (run_imps_results) {
227  if (curr_run_type != RunType::OPTIMIZED_RUN) {
228  parallel_params->AddTestResultsForParallelParams(p_params_curr_run, curr_run_data);
229  }
230  }
231 
232  //set optimized parallel parameters if next run is final run
233  //that uses optimized parallel parameters
234  //optimized parallel parameters are determined from previous evaluation runs
235  //using multiple evaluation parallel parameters
236  if (run_num == (parallel_params_vect.size() - 1)) {
237  parallel_params->SetOptimizedParams();
238  }
239  }
240  }
241 
242  return out_run_data;
243  }
