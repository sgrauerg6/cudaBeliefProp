115  MultRunData::mapped_type RunEvalBenchmark(
116  const run_environment::RunImpSettings& run_imp_settings,
117  bool run_w_loop_iters_templated) const
118  {
119  MultRunData::mapped_type::value_type out_run_data;
120  enum class RunType { ONLY_RUN, OPTIMIZED_RUN, TEST_PARAMS };
121 
122  //set up parallel parameters for specific benchmark
123  std::shared_ptr<ParallelParams> parallel_params =
124  SetUpParallelParams(run_imp_settings);
125 
126  //get constant references to default and optimized options
127  //for parallel parameters
128  const auto& [default_p_params, alt_p_params] =
129  run_imp_settings.p_params_default_alt_options;
130 
131  //generate set with default parallel parameters and all
132  //alternate parallel parameter options
133  auto p_param_options = alt_p_params;
134  p_param_options.insert(default_p_params);
135 
136  //if optimizing parallel parameters, run BP for each parallel parameters option,
137  //retrieve best parameters for each kernel or overall for the run,
138  //and then run BP with best found parallel parameters
139  //if not optimizing parallel parameters, run BP once using default parallel parameters
140  for (auto [p_params_iter, run_num] = std::tuple{p_param_options.cbegin(), 0u};
141  run_num < (p_param_options.size() + 1);
142  p_params_iter++, run_num++)
143  {
144  //initialize current run type to specify if current run is only run, run with
145  //default params, test params run, or final run with optimized params and
146  //get and set parallel parameters for current run if not final run
147  //that uses optimized parameters
148  RunType curr_run_type;
149  if (alt_p_params.size() == 0) {
150  //no alternate parallel parameters to test
151  //so only one run with default parameters
152  curr_run_type = RunType::ONLY_RUN;
153  parallel_params->SetParallelDims(
154  default_p_params);
155  }
156  else if (run_num == p_param_options.size()) {
157  //last run with optimized parallel parameters
158  curr_run_type = RunType::OPTIMIZED_RUN;
159  //optimized parallel parameters are determined from previous evaluation runs
160  //using multiple evaluation parallel parameters
161  parallel_params->SetOptimizedParams();
162  }
163  else {
164  //run with test parallel parameters that will be used to determine
165  //optimized parallel parameters
166  curr_run_type = RunType::TEST_PARAMS;
167  //set parallel parameters to parameters corresponding to current run
168  //for each BP processing level
169  parallel_params->SetParallelDims(*p_params_iter);
170  }
171 
172  //run results are output as part of run evaluation if run is not using
173  //test parallel parameters or is using test parallel parameters that
174  //match default parallel parameters
175  const bool run_results_output{
176  ((curr_run_type != RunType::TEST_PARAMS) ||
177  (*p_params_iter == default_p_params))};
178 
179  //store input params data if current run results are output
180  RunData curr_run_data;
181  if (run_results_output)
182  {
183  //add input and parameters data for specific benchmark to current run data
184  curr_run_data.AddDataWHeader(
185  std::string(run_eval::kInputIdxHeader),
186  std::to_string(NUM_INPUT));
187  curr_run_data.AppendData(
188  InputAndParamsForCurrBenchmark(run_w_loop_iters_templated));
189  if ((alt_p_params.size() > 0) &&
190  (run_imp_settings.opt_parallel_params_setting ==
191  run_environment::OptParallelParamsSetting::kAllowDiffKernelParallelParamsInRun))
192  {
193  //add parallel parameters for each kernel to current input data if
194  //allowing different parallel parameters for each kernel
195  //in the same run
196  curr_run_data.AppendData(parallel_params->AsRunData());
197  }
198  }
199 
200  //run benchmark implementation(s) and return null output if error in run
201  //detailed results stored to file that is generated using stream
202  //only run optimized implementation if run results are not output as part
203  //of evaluation
204  const auto run_imps_results =
205  RunImpsAndCompare(
206  parallel_params,
207  !run_results_output, //if true, only run optimized implementation
208  run_w_loop_iters_templated);
209 
210  //if error in run and run is type where results are output as part of
211  //evaluation, exit function with null output to indicate error
212  if ((!run_imps_results) && run_results_output) {
213  return {};
214  }
215 
216  //add data results from current run if run successful
217  if (run_imps_results) {
218  curr_run_data.AppendData(*run_imps_results);
219  }
220 
221  if (curr_run_type == RunType::TEST_PARAMS) {
222  //current run using test parallel parameters, so add test results
223  //including runtimes for each kernel to parallel parameters object to
224  //use for computation of optimal parallel parameters
225  //if error in run, don't add results for current parallel parameters to
226  //results set
227  if (run_imps_results) {
228  parallel_params->AddTestResultsForParallelParams(*p_params_iter, curr_run_data);
229  }
230  if (*p_params_iter == default_p_params) {
231  //set output for run with default parallel parameters if current test
232  //params runs is run with default parallel parameters
233  out_run_data[run_environment::ParallelParamsSetting::kDefault] = curr_run_data;
234  }
235  }
236  else if (curr_run_type == RunType::ONLY_RUN) {
237  //if current run is only run due to setting of not optimizing parallel parameters,
238  //set optimized and default parallel parameters output to current run data,
239  //resulting in same run data mapped to default and optimized parallel parameters enums
240  out_run_data[run_environment::ParallelParamsSetting::kDefault] = curr_run_data;
241  out_run_data[run_environment::ParallelParamsSetting::kOptimized] = curr_run_data;
242 
243  //exit loop since only running with default parameters and that is done
244  break;
245  }
246  else if (curr_run_type == RunType::OPTIMIZED_RUN) {
247  //set output for final run with optimized parallel parameters
248  out_run_data[run_environment::ParallelParamsSetting::kOptimized] = curr_run_data;
249 
250  //exit loop since run with optimized parameters is final run
251  break;
252  }
253  }
254 
255  return out_run_data;
256  }
