94  MultRunData::mapped_type RunEvalBenchmark(
95  const run_environment::RunImpSettings& run_imp_settings,
96  bool run_w_loop_iters_templated) const
97  {
98  MultRunData::mapped_type::value_type out_run_data;
99  enum class RunType { ONLY_RUN, DEFAULT_PARAMS, OPTIMIZED_RUN, TEST_PARAMS };
100 
101  //set up parallel parameters for specific benchmark
102  std::shared_ptr<ParallelParams> parallel_params = SetUpParallelParams(run_imp_settings);
103 
104  //if optimizing parallel parameters, parallel_params_vect contains parallel parameter settings to run
105  //(and contains only the default parallel parameters if not)
106  std::vector<std::array<unsigned int, 2>> parallel_params_vect{
107  run_imp_settings.opt_parallel_params_setting.first ?
108  run_imp_settings.p_params_default_opt_settings.second :
109  std::vector<std::array<unsigned int, 2>>()};
110 
111  //if optimizing parallel parameters, run BP for each parallel parameters option,
112  //retrieve best parameters for each kernel or overall for the run,
113  //and then run BP with best found parallel parameters
114  //if not optimizing parallel parameters, run BP once using default parallel parameters
115  for (unsigned int run_num = 0; run_num < (parallel_params_vect.size() + 1); run_num++) {
116  //initialize current run type to specify if current run is only run, run with
117  //default params, test params run, or final run with optimized params
118  RunType curr_run_type{RunType::TEST_PARAMS};
119  if (!run_imp_settings.opt_parallel_params_setting.first) {
120  curr_run_type = RunType::ONLY_RUN;
121  }
122  else if (run_num == parallel_params_vect.size()) {
123  curr_run_type = RunType::OPTIMIZED_RUN;
124  }
125 
126  //get and set parallel parameters for current run if not final run that uses optimized parameters
127  std::array<unsigned int, 2> p_params_curr_run{run_imp_settings.p_params_default_opt_settings.first};
128  if (curr_run_type == RunType::ONLY_RUN) {
129  parallel_params->SetParallelDims(run_imp_settings.p_params_default_opt_settings.first);
130  }
131  else if (curr_run_type == RunType::TEST_PARAMS) {
132  //set parallel parameters to parameters corresponding to current run for each BP processing level
133  p_params_curr_run = parallel_params_vect[run_num];
134  parallel_params->SetParallelDims(p_params_curr_run);
135  if (p_params_curr_run == run_imp_settings.p_params_default_opt_settings.first) {
136  //set run type to default parameters if current run uses default parameters
137  curr_run_type = RunType::DEFAULT_PARAMS;
138  }
139  }
140 
141  //store input params data if using default parallel parameters or final run with optimized parameters
142  RunData curr_run_data;
143  if (curr_run_type != RunType::TEST_PARAMS) {
144  //add input and parameters data for specific benchmark to current run data
145  curr_run_data.AddDataWHeader(
146  std::string(run_eval::kInputIdxHeader),
147  std::to_string(NUM_INPUT));
148  curr_run_data.AppendData(InputAndParamsForCurrBenchmark(run_w_loop_iters_templated));
149  if ((run_imp_settings.opt_parallel_params_setting.first) &&
150  (run_imp_settings.opt_parallel_params_setting.second ==
151  run_environment::OptParallelParamsSetting::kAllowDiffKernelParallelParamsInRun))
152  {
153  //add parallel parameters for each kernel to current input data if
154  //allowing different parallel parameters for each kernel
155  //in the same run
156  curr_run_data.AppendData(parallel_params->AsRunData());
157  }
158  }
159 
160  //run only optimized implementation and not single-threaded run if
161  //current run is not final run or is using default parameter parameters
162  const bool run_opt_imp_only{curr_run_type == RunType::TEST_PARAMS};
163 
164  //run benchmark implementation(s) and return null output if error in run
165  //detailed results stored to file that is generated using stream
166  const auto run_imps_results =
167  RunImpsAndCompare(parallel_params, run_opt_imp_only, run_w_loop_iters_templated);
168  curr_run_data.AddDataWHeader(
169  std::string(run_eval::kRunSuccessHeader),
170  run_imps_results.has_value());
171 
172  //if error in run and run is any type other than for testing parameters,
173  //exit function with null output to indicate error
174  if ((!run_imps_results) && (curr_run_type != RunType::TEST_PARAMS)) {
175  return {};
176  }
177 
178  //add data results from current run if run successful
179  if (run_imps_results) {
180  curr_run_data.AppendData(run_imps_results.value());
181  }
182 
183  //add current run results for output if using default parallel
184  //parameters or is final run w/ optimized parallel parameters
185  if (curr_run_type != RunType::TEST_PARAMS) {
186  //set output for runs using default parallel parameters and final run
187  //(which is the same run if not optimizing parallel parameters)
188  if (curr_run_type == RunType::OPTIMIZED_RUN) {
189  out_run_data[run_environment::ParallelParamsSetting::kOptimized] = curr_run_data;
190  }
191  else {
192  out_run_data[run_environment::ParallelParamsSetting::kDefault] = curr_run_data;
193  if (curr_run_type == RunType::ONLY_RUN) {
194  //if current run is only run due to setting of not optimizing parallel parameters,
195  //set optimized parallel parameters output to current run data, resulting in same
196  //run data mapped to default and optimized parallel parameters enums
197  out_run_data[run_environment::ParallelParamsSetting::kOptimized] = curr_run_data;
198  }
199  }
200  }
201 
202  if (run_imp_settings.opt_parallel_params_setting.first) {
203  //retrieve and store results including runtimes for each kernel if
204  //allowing different parallel parameters for each kernel and
205  //total runtime for current run
206  //if error in run, don't add results for current parallel parameters to results set
207  if (run_imps_results) {
208  if (curr_run_type != RunType::OPTIMIZED_RUN) {
209  parallel_params->AddTestResultsForParallelParams(p_params_curr_run, curr_run_data);
210  }
211  }
212 
213  //set optimized parallel parameters if next run is final run
214  //that uses optimized parallel parameters
215  //optimized parallel parameters are determined from previous test runs
216  //using multiple test parallel parameters
217  if (run_num == (parallel_params_vect.size() - 1)) {
218  parallel_params->SetOptimizedParams();
219  }
220  }
221  }
222 
223  return out_run_data;
224  }
