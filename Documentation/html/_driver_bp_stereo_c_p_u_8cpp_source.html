54 void runImp(int argc, char** argv, RunImpSetting impSetting)
55 {
56  //initialize settings to run implementation and evaluation
57  run_environment::RunImpSettings run_imp_settings;
58 
59  //enable optimization of parallel parameters with setting to use the same parallel parameters for all kernels in run
60  //testing on i7-11800H has found that using different parallel parameters (corresponding to OpenMP thread counts)
61  //in different kernels in the optimized CPU implementation can increase runtime (may want to test on additional processors)
62  run_imp_settings.opt_parallel_params_setting =
63  {true,
64  run_environment::OptParallelParamsSetting::kSameParallelParamsAllKernels};
65 
66  //set default parallel parameters and parallel parameters to benchmark when searching for optimal
67  //parallel parameters
68  run_imp_settings.p_params_default_opt_settings =
69  {run_cpu::kParallelParamsDefault,
70  run_cpu::kParallelParameterOptions};
71 
72  //set run name to first argument if it exists
73  //otherwise set to "CurrentRun"
74  run_imp_settings.run_name = (argc > 1) ? argv[1] : "CurrentRun";
75 
76  //adjust thread count to simulate single CPU on dual-CPU system
77  //currently only works as expected if environment variables are set before run such that threads are pinned to socket via
78  //"export OMP_PLACES="sockets"" and "export OMP_PROC_BIND=true" commands on command line
79  if (impSetting == RunImpSetting::kRunImpSimSingleCPUTwoCPUSystem)
80  {
81  //adjust settings to simulate run on single CPU in two-CPU system, specifically set parallel thread count options so that
82  //maximum number of parallel threads is thread count of single CPU and set environment variables so that CPU threads
83  //are pinned to socket
84  //set default parallel threads count to be number of threads on a single CPU in the two-CPU system
85  run_imp_settings.p_params_default_opt_settings.first =
86  {std::thread::hardware_concurrency() / 2, 1};
87 
88  //erase parallel thread count options with more than the number of threads on a single CPU in the two-CPU system
89  run_imp_settings.RemoveParallelParamAboveMaxThreads(
90  std::thread::hardware_concurrency() / 2);
91 
92  //adjust settings so that CPU threads pinned to socket to simulate run on single CPU
93  //TODO: commented out since currently has no effect; needs to be set before run by
94  //calling "export OMP_PLACES="sockets"" and "export OMP_PROC_BIND=true" commands on
95  //command line before run
96  //run_environment::CPUThreadsPinnedToSocket().operator()(true);
97 
98  //append run name to specify that simulating single CPU on dual-CPU system
99  if (run_imp_settings.run_name) {
100  *(run_imp_settings.run_name) += "_SimSingleCPUOnDualCPUSystem";
101  }
102  }
103  //check if running implementation with CPU threads pinned to socket (for cases with multiple CPUs)
104  //TODO: currently not supported due to code for setting CPU threads to be pinned to socket not working as expected
105  //can still run implementation with CPU threads pinned to socket by calling "export OMP_PLACES="sockets"" and
106  //"export OMP_PROC_BIND=true" commands on command line before run
107  /*else if (impSetting == RunImpSetting::kRunImpThreadsPinnedToSocket) {
108  //adjust settings so that CPU threads pinned to socket
109  //TODO: commented out since currently has no effect
110  //run_environment::CPUThreadsPinnedToSocket().operator()(true);
111 
112  //append run name to specify that CPU threads pinned to socket
113  if (run_imp_settings.run_name) {
114  *(run_imp_settings.run_name) += "_ThreadsPinnedToSocket";
115  }
116  }*/
117 
118  //set datatype(s) to use in run processing in evaluation
119  run_imp_settings.datatypes_eval_sizes =
120  {run_eval::kDataTypesEvalSizes.begin(),
121  run_eval::kDataTypesEvalSizes.end()};
122 
123  //set setting of whether or not to use templated loop iterations in implementation
124  //in evaluation runs
125  run_imp_settings.templated_iters_setting =
126  run_eval::kTemplatedItersEvalSettings;
127 
128  //set path of baseline runtimes and baseline description
129  run_imp_settings.baseline_runtimes_path_desc =
130  {beliefprop::kBaselineRunDataPath,
131  beliefprop::kBaselineRunDesc};
132 
133  //set data subsets to evaluate separate from all data
134  run_imp_settings.subset_desc_input_sig = beliefprop::kEvalDataSubsets;
135 
136  //remove any parallel processing below given minimum number of threads
137  run_imp_settings.RemoveParallelParamBelowMinThreads(
138  run_cpu::kMinNumThreadsRun);
139 
140 //run multiple belief propagation across multiple inputs and configurations,
141 //with specific vectorization configurations dependent what's supported on
142 //current CPU as defined at compile time
143 #if (CPU_VECTORIZATION_DEFINE == AVX_512_DEFINE)
144  //run belief propagation with AVX512, AVX256, and no vectorization implementations,
145  //with the AVX512 implementation given first as the expected fastest implementation
146  RunImpMultTypesAccels().operator()({
147  {run_environment::AccSetting::kAVX512,
148  std::make_shared<RunImpMultInputsBp>(run_environment::AccSetting::kAVX512)},
149  {run_environment::AccSetting::kAVX256,
150  std::make_shared<RunImpMultInputsBp>(run_environment::AccSetting::kAVX256)},
151  {run_environment::AccSetting::kNone,
152  std::make_shared<RunImpMultInputsBp>(run_environment::AccSetting::kNone)}},
153  run_imp_settings,
154  std::make_unique<EvaluateImpResultsBp>());
155 #elif (CPU_VECTORIZATION_DEFINE == AVX_256_DEFINE)
156  //run belief propagation with AVX256 and no vectorization implementations,
157  //with the AVX256 implementation given first as the expected fastest implementation
158  RunImpMultTypesAccels().operator()({
159  {run_environment::AccSetting::kAVX256,
160  std::make_shared<RunImpMultInputsBp>(run_environment::AccSetting::kAVX256)},
161  {run_environment::AccSetting::kNone,
162  std::make_shared<RunImpMultInputsBp>(run_environment::AccSetting::kNone)}},
163  run_imp_settings,
164  std::make_unique<EvaluateImpResultsBp>());
165 #elif (CPU_VECTORIZATION_DEFINE == NEON_DEFINE)
166  //run belief propagation with NEON and no vectorization implementations,
167  //with the NEON implementation given first as the expected fastest implementation
168  RunImpMultTypesAccels().operator()({
169  {run_environment::AccSetting::kNEON,
170  std::make_shared<RunImpMultInputsBp>(run_environment::AccSetting::kNEON)},
171  {run_environment::AccSetting::kNone,
172  std::make_shared<RunImpMultInputsBp>(run_environment::AccSetting::kNone)}},
173  run_imp_settings,
174  std::make_unique<EvaluateImpResultsBp>());
175 #endif //CPU_VECTORIZATION_DEFINE
176 }

194 int main(int argc, char** argv)
195 {
196  //if running on a system with two cpus, run implementation with settings adjusted
197  //to simulate run on single CPU if specified in second command line argument
198  if ((argc > 2) && (std::string(argv[2]) == std::string(run_cpu::kSimulateSingleCPU))) {
199  std::cout << "Running optimized CPU implementation with settings adjusted such that "
200  "a single CPU is simulated on a dual-CPU system." << std::endl;
201  std::cout << "Results only as expected if running on dual-CPU system and "
202  "environment variables set so that threads pinned to CPU socket." << std::endl;
203  runImp(argc, argv, RunImpSetting::kRunImpSimSingleCPUTwoCPUSystem);
204  }
205  else {
206  //run default implementation
207  std::cout << "Running optimized CPU implementation" << std::endl;
208  runImp(argc, argv, RunImpSetting::kRunImpDefault);
209  }
210 
211  return 0;
212 }
