132 //retrieve optimized parameters from results across multiple runs with different parallel parameters and set current parameters
133 //to retrieved optimized parameters
134 void BpParallelParams::SetOptimizedParams() {
135  if (opt_parallel_params_setting_ == run_environment::OptParallelParamsSetting::kAllowDiffKernelParallelParamsInRun) {
136  for (unsigned int num_kernel_set = 0; num_kernel_set < parallel_dims_each_kernel_.size(); num_kernel_set++) {
137  //retrieve and set optimized parallel parameters for final run
138  //std::min_element used to retrieve parallel parameters corresponding to lowest runtime from previous runs
139  std::ranges::transform(p_params_to_run_time_each_kernel_[num_kernel_set],
140  parallel_dims_each_kernel_[num_kernel_set].begin(),
141  [](const auto& tDimsToRunTimeCurrLevel) {
142  return (std::ranges::min_element(tDimsToRunTimeCurrLevel,
143  [](const auto& a, const auto& b) { return a.second < b.second; }))->first; });
144  }
145  }
146  else {
147  //set optimized parallel parameters for all kernels to parallel parameters that got the best runtime across all kernels
148  //seems like setting different parallel parameters for different kernels on GPU decrease runtime but increases runtime on CPU
149  const auto best_parallel_params = std::ranges::min_element(
150  p_params_to_run_time_each_kernel_[beliefprop::kNumKernels][0],
151  [](const auto& a, const auto& b) { return a.second < b.second; })->first;
152  SetParallelDims(best_parallel_params);
153  }
