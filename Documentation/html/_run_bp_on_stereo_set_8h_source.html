106 //protected function to set up, run, and evaluate bp processing on target device using pointers to acceleration-specific smooth image,
107 //process BP, and memory management child class objects
108 template<RunData_t T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
109 std::optional<ProcessStereoSetOutput> RunBpOnStereoSet<T, DISP_VALS, ACCELERATION>::ProcessStereoSet(
110  const std::array<std::string, 2>& ref_test_image_path,
111  const beliefprop::BpSettings& alg_settings,
112  const BpOnDevice<T, DISP_VALS, ACCELERATION>& run_bp_on_device) const
113 {
114  //retrieve the images as well as the width and height
115  const std::array<BpImage<unsigned int>, 2> input_images{
116  BpImage<unsigned int>(ref_test_image_path[0]),
117  BpImage<unsigned int>(ref_test_image_path[1])};
118  const std::array<unsigned int, 2> width_height_images{input_images[0].Width(), input_images[0].Height()};
119 
120  //get total number of pixels in input images
121  const unsigned int tot_num_pixels_images{width_height_images[0] * width_height_images[1]};
122 
123  //initialize structures for timing data
124  std::unordered_map<beliefprop::Runtime_Type, std::array<timingType, 2>> runtime_start_end_timings;
125  DetailedTimings detailed_bp_timings(beliefprop::kTimingNames);
126 
127  //initialize output disparity map
128  DisparityMap<float> output_disparity_map(width_height_images);
129 
130  //allocate data for bp processing on target device ahead of runs if option selected
131  T* bp_data{nullptr};
132  T* bp_proc_store{nullptr};
133  if constexpr (beliefprop::kAllocateFreeBpMemoryOutsideRuns) {
134  //allocate memory on device for bp processing
135  const std::size_t num_data = beliefprop::BpLevel::TotalDataForAlignedMemoryAllLevels<T, ACCELERATION>(
136  width_height_images, alg_settings.num_disp_vals, alg_settings.num_levels);
137  bp_data = run_bp_on_device.mem_management_bp_run->AllocateAlignedMemoryOnDevice(10*num_data, ACCELERATION);
138  if (run_bp_on_device.run_bp_stereo->ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) { return {}; }
139 
140  beliefprop::BpLevel bottom_bp_level(width_height_images, 0, 0, ACCELERATION);
141  const std::size_t total_data_bottom_level = bottom_bp_level.NumDataInBpArrays<T>(alg_settings.num_disp_vals);
142  bp_proc_store = run_bp_on_device.mem_management_bp_run->AllocateAlignedMemoryOnDevice(total_data_bottom_level, ACCELERATION);
143  if (run_bp_on_device.run_bp_stereo->ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) { return {}; }
144  }
145 
146  //run bp processing for specified number of runs
147  for (unsigned int num_run = 0; num_run < beliefprop::kNumBpStereoRuns; num_run++)
148  {
149  //allocate the device memory to store and x and y smoothed images
150  std::array<float*, 2> smoothed_images{
151  run_bp_on_device.mem_management_images->AllocateMemoryOnDevice(tot_num_pixels_images),
152  run_bp_on_device.mem_management_images->AllocateMemoryOnDevice(tot_num_pixels_images)};
153 
154  if (run_bp_on_device.run_bp_stereo->ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
155  //return std::optional object with no value indicating error in run
156  return {};
157  }
158 
159  //set start timer for specified runtime segments at time before smoothing images
160  runtime_start_end_timings[beliefprop::Runtime_Type::kSmoothing][0] = std::chrono::system_clock::now();
161  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalNoTransfer][0] = std::chrono::system_clock::now();
162  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalWithTransfer][0] = std::chrono::system_clock::now();
163 
164  //first smooth the images using the Gaussian filter with the given smoothing sigma value
165  //smoothed images are stored on the target device
166  for (unsigned int i = 0; i < 2; i++) {
167  (*(run_bp_on_device.smooth_image))(input_images[i], alg_settings.smoothing_sigma, smoothed_images[i]);
168  if (run_bp_on_device.run_bp_stereo->ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
169  //return std::optional object with no value indicating error in run
170  return {};
171  }
172  }
173 
174  //end timer for image smoothing and add to image smoothing timings
175  runtime_start_end_timings[beliefprop::Runtime_Type::kSmoothing][1] = std::chrono::system_clock::now();
176 
177  //get and store time point at start of bp processing
178  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalBp][0] = std::chrono::system_clock::now();
179 
180  //run belief propagation on device as specified by input pointer to ProcessBp object run_bp_stereo
181  //returns detailed timings for bp run
182  auto bp_stereo_output = (*(run_bp_on_device.run_bp_stereo))(
183  smoothed_images, alg_settings, width_height_images,
184  bp_data, bp_proc_store, run_bp_on_device.mem_management_bp_run);
185  if (!bp_stereo_output) {
186  //return std::optional object with no value if null output indicating error in run
187  return {};
188  }
189 
190  //get and store end timepoint of bp run for computation of total runtime
191  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalBp][1] =
192  std::chrono::system_clock::now();
193  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalNoTransfer][1] =
194  std::chrono::system_clock::now();
195 
196  //transfer the disparity map estimation on the device to the host for output
197  run_bp_on_device.mem_management_images->TransferDataFromDeviceToHost(
198  output_disparity_map.PointerToPixelsStart(), bp_stereo_output->first, tot_num_pixels_images);
199 
200  //compute timings for each portion of interest and add to vector timings
201  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalWithTransfer][1] =
202  std::chrono::system_clock::now();
203 
204  //retrieve the duration for each segment and add to bp timings
205  std::ranges::for_each(runtime_start_end_timings,
206  [&detailed_bp_timings] (const auto& current_runtime_name_timing) {
207  detailed_bp_timings.AddTiming(current_runtime_name_timing.first,
208  current_runtime_name_timing.second[1] - current_runtime_name_timing.second[0]);
209  });
210 
211  //add bp timings from current run to overall timings
212  detailed_bp_timings.AddToCurrentTimings(bp_stereo_output->second);
213 
214  //free the space allocated to the resulting disparity map and smoothed images on the computation device
215  run_bp_on_device.mem_management_images->FreeMemoryOnDevice(bp_stereo_output->first);
216  for (auto& smoothed_image : smoothed_images) {
217  run_bp_on_device.mem_management_images->FreeMemoryOnDevice(smoothed_image);
218  }
219  }
220 
221  //free data for bp processing on target device if this memory
222  //management set to be done outside of runs
223  if constexpr (beliefprop::kAllocateFreeBpMemoryOutsideRuns) {
224  run_bp_on_device.mem_management_bp_run->FreeAlignedMemoryOnDevice(bp_data);
225  run_bp_on_device.mem_management_bp_run->FreeAlignedMemoryOnDevice(bp_proc_store);
226  }
227 
228  //construct RunData object with bp input and timing info
229  RunData run_data;
230  run_data.AddDataWHeader(std::string(beliefprop::kImageWidthHeader), width_height_images[0]);
231  run_data.AddDataWHeader(std::string(beliefprop::kImageHeightHeader), width_height_images[1]);
232  run_data.AppendData(detailed_bp_timings.AsRunData());
233 
234  //construct and return ProcessStereoSetOutput object inside of std::optional object
235  return {ProcessStereoSetOutput{
236  detailed_bp_timings.MedianTiming(
237  beliefprop::Runtime_Type::kTotalWithTransfer), std::move(output_disparity_map), run_data}};
238 }
