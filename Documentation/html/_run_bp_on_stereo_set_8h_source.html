128 //protected function to set up, run, and evaluate bp processing on target device using pointers to acceleration-specific smooth image,
129 //process BP, and memory management child class objects
130 template<RunData_t T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
131 std::optional<beliefprop::BpRunOutput> RunBpOnStereoSet<T, DISP_VALS, ACCELERATION>::ProcessStereoSet(
132  const std::array<std::string, 2>& ref_test_image_path,
133  const beliefprop::BpSettings& alg_settings,
134  const beliefprop::BpOnDevice<T, DISP_VALS, ACCELERATION>& run_bp_on_device) const
135 {
136  //retrieve the images as well as the width and height
137  const std::array<BpImage<unsigned int>, 2> input_images{
138  BpImage<unsigned int>(ref_test_image_path[0]),
139  BpImage<unsigned int>(ref_test_image_path[1])};
140  const std::array<unsigned int, 2> width_height_images{input_images[0].Width(), input_images[0].Height()};
141 
142  //get total number of pixels in input images
143  const unsigned int tot_num_pixels_images{width_height_images[0] * width_height_images[1]};
144 
145  //initialize structures for timing data
146  std::unordered_map<beliefprop::Runtime_Type, std::array<timingType, 2>> runtime_start_end_timings;
147  DetailedTimings detailed_bp_timings(beliefprop::kTimingNames);
148 
149  //initialize output disparity map
150  DisparityMap<float> output_disparity_map(width_height_images);
151 
152  //allocate data for bp processing on target device ahead of runs if option selected
153  T* bp_data{nullptr};
154  T* bp_proc_store{nullptr};
155  if constexpr (beliefprop::kAllocateFreeBpMemoryOutsideRuns) {
156  //allocate memory on device for bp processing
157  const std::size_t num_data = BpLevel::TotalDataForAlignedMemoryAllLevels<T, ACCELERATION>(
158  width_height_images, alg_settings.num_disp_vals, alg_settings.num_levels);
159  bp_data = run_bp_on_device.mem_management_bp_run->AllocateAlignedMemoryOnDevice(10*num_data, ACCELERATION);
160  if (run_bp_on_device.run_bp_stereo->ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) { return {}; }
161 
162  BpLevel bottom_bp_level(width_height_images, 0, 0, ACCELERATION);
163  const std::size_t total_data_bottom_level = bottom_bp_level.NumDataInBpArrays<T>(alg_settings.num_disp_vals);
164  bp_proc_store = run_bp_on_device.mem_management_bp_run->AllocateAlignedMemoryOnDevice(total_data_bottom_level, ACCELERATION);
165  if (run_bp_on_device.run_bp_stereo->ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) { return {}; }
166  }
167 
168  //run bp processing for specified number of runs
169  const unsigned int num_evaluation_runs{beliefprop::NumBpStereoRuns(alg_settings.num_disp_vals)};
170  for (unsigned int num_run = 0;
171  num_run < num_evaluation_runs;
172  num_run++)
173  {
174  //allocate the device memory to store and x and y smoothed images
175  std::array<float*, 2> smoothed_images{
176  run_bp_on_device.mem_management_images->AllocateMemoryOnDevice(tot_num_pixels_images),
177  run_bp_on_device.mem_management_images->AllocateMemoryOnDevice(tot_num_pixels_images)};
178 
179  if (run_bp_on_device.run_bp_stereo->ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
180  //return std::optional object with no value indicating error in run
181  return {};
182  }
183 
184  //set start timer for specified runtime segments at time before smoothing images
185  runtime_start_end_timings[beliefprop::Runtime_Type::kSmoothing][0] = std::chrono::system_clock::now();
186  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalNoTransfer][0] = std::chrono::system_clock::now();
187  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalWithTransfer][0] = std::chrono::system_clock::now();
188 
189  //first smooth the images using the Gaussian filter with the given smoothing sigma value
190  //smoothed images are stored on the target device
191  for (unsigned int i = 0; i < 2; i++) {
192  (*(run_bp_on_device.smooth_image))(input_images[i], alg_settings.smoothing_sigma, smoothed_images[i]);
193  if (run_bp_on_device.run_bp_stereo->ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
194  //return std::optional object with no value indicating error in run
195  return {};
196  }
197  }
198 
199  //end timer for image smoothing and add to image smoothing timings
200  runtime_start_end_timings[beliefprop::Runtime_Type::kSmoothing][1] = std::chrono::system_clock::now();
201 
202  //get and store time point at start of bp processing
203  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalBp][0] = std::chrono::system_clock::now();
204 
205  //run belief propagation on device as specified by input pointer to ProcessBp object run_bp_stereo
206  //returns detailed timings for bp run
207  auto bp_stereo_output = (*(run_bp_on_device.run_bp_stereo))(
208  smoothed_images, alg_settings, width_height_images,
209  bp_data, bp_proc_store, run_bp_on_device.mem_management_bp_run);
210  if (!bp_stereo_output) {
211  //return std::optional object with no value if null output indicating error in run
212  return {};
213  }
214 
215  //get and store end timepoint of bp run for computation of total runtime
216  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalBp][1] =
217  std::chrono::system_clock::now();
218  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalNoTransfer][1] =
219  std::chrono::system_clock::now();
220 
221  //transfer the disparity map estimation on the device to the host for output
222  run_bp_on_device.mem_management_images->TransferDataFromDeviceToHost(
223  output_disparity_map.PointerToPixelsStart(), bp_stereo_output->first, tot_num_pixels_images);
224 
225  //compute timings for each portion of interest and add to vector timings
226  runtime_start_end_timings[beliefprop::Runtime_Type::kTotalWithTransfer][1] =
227  std::chrono::system_clock::now();
228 
229  //retrieve the duration for each segment and add to bp timings
230  std::ranges::for_each(runtime_start_end_timings,
231  [&detailed_bp_timings] (const auto& current_runtime_name_timing) {
232  detailed_bp_timings.AddTiming(current_runtime_name_timing.first,
233  current_runtime_name_timing.second[1] - current_runtime_name_timing.second[0]);
234  });
235 
236  //add bp timings from current run to overall timings
237  detailed_bp_timings.AddToCurrentTimings(bp_stereo_output->second);
238 
239  //free the space allocated to the resulting disparity map and smoothed images on the computation device
240  run_bp_on_device.mem_management_images->FreeMemoryOnDevice(bp_stereo_output->first);
241  for (auto& smoothed_image : smoothed_images) {
242  run_bp_on_device.mem_management_images->FreeMemoryOnDevice(smoothed_image);
243  }
244  }
245 
246  //free data for bp processing on target device if this memory
247  //management set to be done outside of runs
248  if constexpr (beliefprop::kAllocateFreeBpMemoryOutsideRuns) {
249  run_bp_on_device.mem_management_bp_run->FreeAlignedMemoryOnDevice(bp_data);
250  run_bp_on_device.mem_management_bp_run->FreeAlignedMemoryOnDevice(bp_proc_store);
251  }
252 
253  //construct RunData object with bp input and timing info
254  RunData run_data;
255  run_data.AddDataWHeader(std::string(beliefprop::kImageWidthHeader), width_height_images[0]);
256  run_data.AddDataWHeader(std::string(beliefprop::kImageHeightHeader), width_height_images[1]);
257  run_data.AddDataWHeader(std::string(beliefprop::kNumEvalRuns), num_evaluation_runs);
258  run_data.AppendData(detailed_bp_timings.AsRunData());
259 
260  //construct and return beliefprop::BpRunOutput object inside of std::optional object
261  return {beliefprop::BpRunOutput{
262  detailed_bp_timings.MedianTiming(
263  beliefprop::Runtime_Type::kTotalWithTransfer), std::move(output_disparity_map), run_data}};
264 }
