294  virtual std::pair<beliefprop::DataCostsCheckerboards<T*>, beliefprop::CheckerboardMessages<T*>>
295  OrganizeDataCostsAndMessageDataAllLevels(
296  T* data_all_levels, std::size_t num_data_allocate_per_data_costs_message_data_array) const
297  {
298  beliefprop::DataCostsCheckerboards<T*> data_costs_device_checkerboard_all_levels;
299  data_costs_device_checkerboard_all_levels[0] = data_all_levels;
300  data_costs_device_checkerboard_all_levels[1] =
301  &(data_costs_device_checkerboard_all_levels[0][1 * (num_data_allocate_per_data_costs_message_data_array)]);
302 
303  beliefprop::CheckerboardMessages<T*> messages_device_all_levels;
304  for (const auto checkerboard_num : {static_cast<unsigned int>(beliefprop::CheckerboardPart::kCheckerboardPart0),
305  static_cast<unsigned int>(beliefprop::CheckerboardPart::kCheckerboardPart1)})
306  {
307  for (unsigned int i = 0; i < messages_device_all_levels[checkerboard_num].size(); i++) {
308  messages_device_all_levels[checkerboard_num][i] =
309  &(data_costs_device_checkerboard_all_levels[0][
310  ((checkerboard_num * messages_device_all_levels[checkerboard_num].size()) + (i + 2)) *
311  (num_data_allocate_per_data_costs_message_data_array)]);
312  }
313  }
314 
315  return {data_costs_device_checkerboard_all_levels, messages_device_all_levels};
316  }

347 //run the belief propagation algorithm with on a set of stereo images to generate a disparity map on target device
348 //input is images on target device for computation
349 //output is disparity map and processing runtimes
350 template<RunData_t T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
351 std::optional<std::pair<float*, DetailedTimings<beliefprop::Runtime_Type>>>
352  ProcessBp<T, DISP_VALS, ACCELERATION>::operator()(
353  const std::array<float*, 2> & images_target_device, const beliefprop::BpSettings& alg_settings,
354  const std::array<unsigned int, 2>& width_height_images, T* allocated_mem_bp_processing, T* allocated_memory,
355  const std::unique_ptr<MemoryManagement<T>>& mem_management_bp_run) const
356 {
357  if (ErrorCheck() != run_eval::Status::kNoError) { return {}; }
358 
359  std::unordered_map<beliefprop::Runtime_Type, std::array<timingType, 2>> start_end_times;
360  std::vector<std::array<timingType, 2>> data_costs_timings(alg_settings.num_levels);
361  std::vector<std::array<timingType, 2>> bp_timings(alg_settings.num_levels);
362  std::vector<std::array<timingType, 2>> data_copy_timings(alg_settings.num_levels);
363  std::chrono::duration<double> total_time_bp_iters{0}, total_time_copy_data{0}, total_time_copy_data_kernel{0};
364 
365  //start at the "bottom level" and work way up to determine amount of space needed to store data costs
366  std::vector<beliefprop::BpLevel> bp_levels;
367  bp_levels.reserve(alg_settings.num_levels);
368 
369  //set level properties for bottom level that include processing of full image width/height
370  bp_levels.push_back(beliefprop::BpLevel(width_height_images, 0, 0, ACCELERATION));
371 
372  //compute level properties which includes offset for each data/message array for each level after the bottom level
373  for (unsigned int level_num = 1; level_num < alg_settings.num_levels; level_num++) {
374  //get current level properties from previous level properties
375  bp_levels.push_back(bp_levels[level_num-1].NextBpLevel<T>(alg_settings.num_disp_vals));
376  }
377 
378  start_end_times[beliefprop::Runtime_Type::kInitSettingsMalloc][0] = std::chrono::system_clock::now();
379 
380  //declare and allocate the space on the device to store the data cost component at each
381  //possible movement at each level of the "pyramid" as well as the message data used for bp processing
382  //each checkerboard holds half of the data and checkerboard 0 includes the pixel in slot (0, 0)
383  beliefprop::DataCostsCheckerboards<T*> data_costs_device_all_levels;
384  beliefprop::CheckerboardMessages<T*> messages_device_all_levels;
385 
386  //data for each array at all levels set to data up to final level
387  //(corresponds to offset at final level) plus data amount at final level
388  const std::size_t data_all_levels_each_data_message_arr =
389  bp_levels[alg_settings.num_levels-1].LevelProperties().offset_into_arrays_ +
390  bp_levels[alg_settings.num_levels-1].NumDataInBpArrays<T>(alg_settings.num_disp_vals);
391 
392  //assuming that width includes padding
393  if constexpr (beliefprop::kUseOptGPUMemManagement) {
394  if constexpr (beliefprop::kAllocateFreeBpMemoryOutsideRuns) {
395  std::tie(data_costs_device_all_levels, messages_device_all_levels) =
396  OrganizeDataCostsAndMessageDataAllLevels(
397  allocated_mem_bp_processing, data_all_levels_each_data_message_arr);
398  }
399  else {
400  //call function that allocates all data in single array and then
401  //set offsets in array for data costs and message data locations
402  std::tie(data_costs_device_all_levels, messages_device_all_levels) =
403  AllocateAndOrganizeDataCostsAndMessageDataAllLevels(
404  data_all_levels_each_data_message_arr, mem_management_bp_run);
405  }
406  }
407  else {
408  data_costs_device_all_levels = AllocateMemoryForDataCosts(data_all_levels_each_data_message_arr, mem_management_bp_run);
409  }
410 
411  auto curr_time = std::chrono::system_clock::now();
412  start_end_times[beliefprop::Runtime_Type::kInitSettingsMalloc][1] = curr_time;
413  data_costs_timings[0][0] = curr_time;
414 
415  //initialize the data cost at the bottom level
416  auto error_code = InitializeDataCosts(alg_settings, bp_levels[0], images_target_device, data_costs_device_all_levels);
417  if (error_code != run_eval::Status::kNoError) { return {}; }
418 
419  curr_time = std::chrono::system_clock::now();
420  data_costs_timings[0][1] = curr_time;
421  start_end_times[beliefprop::Runtime_Type::kDataCostsHigherLevel][0] = curr_time;
422 
423  //set the data costs at each level from the bottom level "up"
424  for (unsigned int level_num = 1u; level_num < alg_settings.num_levels; level_num++)
425  {
426  data_costs_timings[level_num][0] = std::chrono::system_clock::now();
427  error_code = InitializeDataCurrentLevel(bp_levels[level_num], bp_levels[level_num - 1],
428  RetrieveLevelDataCosts(data_costs_device_all_levels, bp_levels[level_num - 1u].LevelProperties().offset_into_arrays_),
429  RetrieveLevelDataCosts(data_costs_device_all_levels, bp_levels[level_num].LevelProperties().offset_into_arrays_),
430  alg_settings.num_disp_vals);
431  if (error_code != run_eval::Status::kNoError) { return {}; }
432 
433  data_costs_timings[level_num][1] = std::chrono::system_clock::now();
434  }
435 
436  curr_time = data_costs_timings[alg_settings.num_levels-1][1];
437  start_end_times[beliefprop::Runtime_Type::kDataCostsHigherLevel][1] = curr_time;
438  start_end_times[beliefprop::Runtime_Type::kInitMessages][0] = curr_time;
439 
440  //get and use offset into data at current processing level of pyramid
441  beliefprop::DataCostsCheckerboards<T*> data_costs_device_current_level = RetrieveLevelDataCosts(
442  data_costs_device_all_levels, bp_levels[alg_settings.num_levels - 1u].LevelProperties().offset_into_arrays_);
443 
444  //declare the space for two levels of BP messages
445  //need to support two "sets" of checkerboards because the message data at the current level
446  //in the image pyramid need copied to a lower level without overwriting values
447  std::array<beliefprop::CheckerboardMessages<T*>, 2> messages_curr_next_level;
448 
449  //alternate between indices 0 and 1 for current and next level messages sets
450  //in messages for bp level array
451  unsigned int current_level_messages_idx{0};
452  unsigned int next_level_messages_idx{(current_level_messages_idx + 1) % 2};
453 
454  //assuming that width includes padding
455  if constexpr (beliefprop::kUseOptGPUMemManagement) {
456  messages_curr_next_level[current_level_messages_idx] = RetrieveLevelMessageData(
457  messages_device_all_levels, bp_levels[alg_settings.num_levels - 1u].LevelProperties().offset_into_arrays_);
458  }
459  else {
460  //allocate the space for the message values in the first checkboard set at the current level
461  messages_curr_next_level[current_level_messages_idx] = AllocateMemoryForCheckerboardMessages(
462  bp_levels[alg_settings.num_levels - 1u].NumDataInBpArrays<T>(alg_settings.num_disp_vals), mem_management_bp_run);
463  }
464 
465  start_end_times[beliefprop::Runtime_Type::kInitMessagesKernel][0] = std::chrono::system_clock::now();
466 
467  //initialize all the BP message values at every pixel for every disparity to 0
468  error_code = InitializeMessageValsToDefault(
469  bp_levels[alg_settings.num_levels - 1],
470  messages_curr_next_level[current_level_messages_idx],
471  alg_settings.num_disp_vals);
472  if (error_code != run_eval::Status::kNoError) { return {}; }
473 
474  curr_time = std::chrono::system_clock::now();
475  start_end_times[beliefprop::Runtime_Type::kInitMessagesKernel][1] = curr_time;
476  start_end_times[beliefprop::Runtime_Type::kInitMessages][1] = curr_time;
477 
478  //run BP at each level in the "pyramid" starting on top and continuing to the bottom
479  //where the final movement values are computed...the message values are passed from
480  //the upper level to the lower levels; this pyramid methods causes the BP message values
481  //to converge more quickly
482  for (int level_num = (int)alg_settings.num_levels - 1; level_num >= 0; level_num--)
483  {
484  const auto bp_iter_start_time = std::chrono::system_clock::now();
485 
486  //alternate which checkerboard messages set to use for each level
487  //to support copying from one to the other in parallel
488  //without read-write conflicts
489  error_code = RunBPAtCurrentLevel(
490  alg_settings, bp_levels[(unsigned int)level_num],
491  data_costs_device_current_level,
492  messages_curr_next_level[current_level_messages_idx],
493  allocated_memory);
494  if (error_code != run_eval::Status::kNoError) { return {}; }
495 
496  //retrieve and update total run time for bp processing at levels
497  const auto bp_iter_end_time = std::chrono::system_clock::now();
498  total_time_bp_iters += bp_iter_end_time - bp_iter_start_time;
499  const auto copy_message_values_start_time = std::chrono::system_clock::now();
500 
501  //if not at the "bottom level" copy the current message values at the current level to the corresponding slots next level
502  if (level_num > 0)
503  {
504  //use offset into allocated memory at next level
505  data_costs_device_current_level =
506  RetrieveLevelDataCosts(data_costs_device_all_levels, bp_levels[level_num - 1].LevelProperties().offset_into_arrays_);
507 
508  //assuming that width includes padding
509  if constexpr (beliefprop::kUseOptGPUMemManagement) {
510  messages_curr_next_level[next_level_messages_idx] = RetrieveLevelMessageData(
511  messages_device_all_levels, bp_levels[level_num - 1].LevelProperties().offset_into_arrays_);
512  }
513  else {
514  //allocate space for the message values in the next level
515  messages_curr_next_level[next_level_messages_idx] = AllocateMemoryForCheckerboardMessages(
516  bp_levels[level_num - 1].NumDataInBpArrays<T>(alg_settings.num_disp_vals));
517  }
518 
519  const auto copy_message_values_kernel_start_time = std::chrono::system_clock::now();
520 
521  //copying messages values from messages_curr_next_level[current_level_messages_idx] to
522  //messages_curr_next_level[next_levels_messages_idx]
523  //current_level_messages_idx is later set to next_levels_messages_idx when processing next level
524  error_code = CopyMessageValuesToNextLevelDown(bp_levels[level_num], bp_levels[level_num - 1],
525  messages_curr_next_level[current_level_messages_idx], messages_curr_next_level[next_level_messages_idx],
526  alg_settings.num_disp_vals);
527  if (error_code != run_eval::Status::kNoError) { return {}; }
528 
529  //get timing of copying bp message values to next level
530  const auto copy_message_values_kernel_end_time = std::chrono::system_clock::now();
531  total_time_copy_data_kernel += copy_message_values_kernel_end_time - copy_message_values_kernel_start_time;
532  data_copy_timings[level_num][0] = copy_message_values_kernel_start_time;
533  data_copy_timings[level_num][1] = copy_message_values_kernel_end_time;
534 
535  //assuming that width includes padding
536  if constexpr (!beliefprop::kUseOptGPUMemManagement) {
537  //free the now-copied from computed data of the completed level
538  FreeCheckerboardMessagesMemory(messages_curr_next_level[current_level_messages_idx], mem_management_bp_run);
539  }
540 
541  //update current and next level messages indices for processing at next level
542  //specifically indices of each which are 0 and 1 are swapped with each other
543  current_level_messages_idx = (current_level_messages_idx + 1) % 2;
544  next_level_messages_idx = (current_level_messages_idx + 1) % 2;
545  }
546 
547  //update total timing for copying messages and bp timing at level
548  const auto copy_message_values_end_time = std::chrono::system_clock::now();
549  total_time_copy_data += copy_message_values_end_time - copy_message_values_start_time;
550  bp_timings[level_num][0] = bp_iter_start_time;
551  bp_timings[level_num][1] = bp_iter_end_time;
552  }
553 
554  start_end_times[beliefprop::Runtime_Type::kOutputDisparity][0] = std::chrono::system_clock::now();
555 
556  //assume in bottom level when retrieving output disparity
557  float* result_disp_map_device = RetrieveOutputDisparity(bp_levels[0],
558  data_costs_device_current_level,
559  messages_curr_next_level[current_level_messages_idx],
560  alg_settings.num_disp_vals);
561  if (result_disp_map_device == nullptr) { return {}; }
562 
563  curr_time = std::chrono::system_clock::now();
564  start_end_times[beliefprop::Runtime_Type::kOutputDisparity][1] = curr_time;
565  start_end_times[beliefprop::Runtime_Type::kFinalFree][0] = curr_time;
566 
567  if constexpr (beliefprop::kUseOptGPUMemManagement) {
568  if constexpr (beliefprop::kAllocateFreeBpMemoryOutsideRuns) {
569  //do nothing; memory freed outside of runs
570  }
571  else {
572  //now free the allocated data space; all data in single array when
573  //beliefprop::kUseOptGPUMemManagement set to true
574  FreeDataCostsAllDataInSingleArray(data_costs_device_all_levels, mem_management_bp_run);
575  }
576  }
577  else {
578  //free the device storage allocated to the message values used to retrieve the output movement values
579  FreeCheckerboardMessagesMemory(messages_curr_next_level[current_level_messages_idx], mem_management_bp_run);
580 
581  //now free the allocated data space
582  FreeDataCostsMemory(data_costs_device_all_levels, mem_management_bp_run);
583  }
584 
585  start_end_times[beliefprop::Runtime_Type::kFinalFree][1] = std::chrono::system_clock::now();
586 
587  //retrieve and store processing runtimes for bp at each level
588  //currently a maximum of 10 levels is supported
589  start_end_times[beliefprop::Runtime_Type::kLevel0DataCosts] = data_costs_timings[0];
590  start_end_times[beliefprop::Runtime_Type::kLevel0Bp] = bp_timings[0];
591  start_end_times[beliefprop::Runtime_Type::kLevel0Copy] = data_copy_timings[0];
592  if (bp_timings.size() > 1) {
593  start_end_times[beliefprop::Runtime_Type::kLevel1DataCosts] = data_costs_timings[1];
594  start_end_times[beliefprop::Runtime_Type::kLevel1Bp] = bp_timings[1];
595  start_end_times[beliefprop::Runtime_Type::kLevel1Copy] = data_copy_timings[1];
596  }
597  if (bp_timings.size() > 2) {
598  start_end_times[beliefprop::Runtime_Type::kLevel2DataCosts] = data_costs_timings[2];
