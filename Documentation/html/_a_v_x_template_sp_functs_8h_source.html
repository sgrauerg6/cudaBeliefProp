44 short util_functs::ZeroVal<short>()
45 {
46 #ifdef _WIN32
47  __m128 dataInAvxReg = _mm_set_ss(0.0);
48  __m128i convertedData = _mm_cvtps_ph(dataInAvxReg, 0);

50 #else
51  return _cvtss_sh(0.0f, 0);

53 }
