48  inline RunData retrieveDeviceProperties(int num_device)
49  {
50  cudaDeviceProp prop;
51  std::array<int, 2> cuda_version_driver_runtime;
52  cudaGetDeviceProperties(&prop, num_device);
53  cudaDriverGetVersion(&(cuda_version_driver_runtime[0]));
54  cudaRuntimeGetVersion(&(cuda_version_driver_runtime[1]));
55 
56  RunData run_data;
57  run_data.AddDataWHeader("Device " + std::to_string(num_device),
58  std::string(prop.name) + " with " + std::to_string(prop.multiProcessorCount) +
59  " multiprocessors");
60  run_data.AddDataWHeader(std::string(beliefprop::kCUDAVersionHeader),
61  std::to_string(cuda_version_driver_runtime[0]));
62  run_data.AddDataWHeader(std::string(beliefprop::kCUDARuntimeHeader),
63  std::to_string(cuda_version_driver_runtime[1]));
64  return run_data;
65  }
