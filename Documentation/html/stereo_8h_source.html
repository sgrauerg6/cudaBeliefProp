66 inline void RunBpStereoCPUSingleThread<T, DISP_VALS, ACCELERATION>::dt(float f[DISP_VALS]) const {
67  for (unsigned int q = 1; q < DISP_VALS; q++) {
68  float prev = f[q - 1] + 1.0F;
69  if (prev < f[q])
70  f[q] = prev;
71  }
72  for (int q = (int)DISP_VALS - 2; q >= 0; q--) {
73  float prev = f[q + 1] + 1.0F;
74  if (prev < f[q])
75  f[q] = prev;
76  }
77 }

80 template<typename T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
81 inline void RunBpStereoCPUSingleThread<T, DISP_VALS, ACCELERATION>::msg(float s1[DISP_VALS],
82  float s2[DISP_VALS], float s3[DISP_VALS],
83  float s4[DISP_VALS], float dst[DISP_VALS],
84  float disc_k_bp) const {
85  float val;
86 
87  // aggregate and find min
88  float minimum = beliefprop::kInfBp;
89  for (unsigned int value = 0; value < DISP_VALS; value++) {
90  dst[value] = s1[value] + s2[value] + s3[value] + s4[value];
91  if (dst[value] < minimum)
92  minimum = dst[value];
93  }
94 
95  // dt
96  dt(dst);
97 
98  // truncate
99  minimum += disc_k_bp;
100  for (unsigned int value = 0; value < DISP_VALS; value++)
101  if (minimum < dst[value])
102  dst[value] = minimum;
103 
104  // normalize
105  val = 0;
106  for (unsigned int value = 0; value < DISP_VALS; value++)
107  val += dst[value];
108 
109  val /= DISP_VALS;
110  for (unsigned int value = 0; value < DISP_VALS; value++)
111  dst[value] -= val;
112 }

115 template<typename T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
116 inline image<float[DISP_VALS]> * RunBpStereoCPUSingleThread<T, DISP_VALS, ACCELERATION>::comp_data(
117  image<uchar> *img1, image<uchar> *img2, const beliefprop::BpSettings& alg_settings) const {
118  unsigned int width{(unsigned int)img1->width()};
119  unsigned int height{(unsigned int)img1->height()};
120  image<float[DISP_VALS]> *data = new image<float[DISP_VALS]>(width, height);
121 
122  image<float> *sm1, *sm2;
123  if (alg_settings.smoothing_sigma >= 0.1) {
124  sm1 = FilterImage::smooth(img1, alg_settings.smoothing_sigma);
125  sm2 = FilterImage::smooth(img2, alg_settings.smoothing_sigma);
126  } else {
127  sm1 = imageUCHARtoFLOAT(img1);
128  sm2 = imageUCHARtoFLOAT(img2);
129  }
130 
131  for (unsigned int y = 0; y < height; y++) {
132  for (unsigned int x = DISP_VALS - 1; x < width; x++) {
133  for (unsigned int value = 0; value < DISP_VALS; value++) {
134  const float val = abs(imRef(sm1, x, y) - imRef(sm2, x - value, y));
135  imRef(data, x, y)[value] = alg_settings.lambda_bp * std::min(val, alg_settings.data_k_bp);
136  }
137  }
138  }
139 
140  delete sm1;
141  delete sm2;
142  return data;
143 }

146 template<typename T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
147 inline image<uchar> * RunBpStereoCPUSingleThread<T, DISP_VALS, ACCELERATION>::output(image<float[DISP_VALS]> *u,
148  image<float[DISP_VALS]> *d, image<float[DISP_VALS]> *l,
149  image<float[DISP_VALS]> *r, image<float[DISP_VALS]> *data) const {
150  unsigned int width{(unsigned int)data->width()};
151  unsigned int height{(unsigned int)data->height()};
152  image<uchar> *out = new image<uchar>(width, height);
153 
154  for (unsigned int y = 1; y < height - 1; y++) {
155  for (unsigned int x = 1; x < width - 1; x++) {
156  // keep track of best value for current pixel
157  unsigned int best = 0;
158  float best_val = beliefprop::kInfBp;
159  for (unsigned int value = 0; value < DISP_VALS; value++) {
160  const float val =
161  imRef(u, x, y+1)[value] +
162  imRef(d, x, y-1)[value] +
163  imRef(l, x+1, y)[value] +
164  imRef(r, x-1, y)[value] +
165  imRef(data, x, y)[value];
166 
167  if (val < best_val) {
168  best_val = val;
169  best = value;
170  }
171  }
172  imRef(out, x, y) = best;
173  }
174  }
175 
176  return out;
177 }

211 template<typename T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
212 inline std::pair<image<uchar>*, RunData> RunBpStereoCPUSingleThread<T, DISP_VALS, ACCELERATION>::stereo_ms(image<uchar> *img1, image<uchar> *img2,
213  const beliefprop::BpSettings& alg_settings, std::chrono::duration<double>& runtime) const {
214  image<float[DISP_VALS]> *u[alg_settings.num_levels];
215  image<float[DISP_VALS]> *d[alg_settings.num_levels];
216  image<float[DISP_VALS]> *l[alg_settings.num_levels];
217  image<float[DISP_VALS]> *r[alg_settings.num_levels];
218  image<float[DISP_VALS]> *data[alg_settings.num_levels];
219 
220  auto timeStart = std::chrono::system_clock::now();
221 
222  // data costs
223  data[0] = comp_data(img1, img2, alg_settings);
224 
225  // data pyramid
226  for (unsigned int i = 1; i < alg_settings.num_levels; i++) {
227  const unsigned int old_width = (unsigned int)data[i - 1]->width();
228  const unsigned int old_height = (unsigned int)data[i - 1]->height();
229  const unsigned int new_width = (unsigned int)ceil(old_width / 2.0);
230  const unsigned int new_height = (unsigned int)ceil(old_height / 2.0);
231 
232  assert(new_width >= 1);
233  assert(new_height >= 1);
234 
235  data[i] = new image<float[DISP_VALS]>(new_width, new_height);
236  for (unsigned int y = 0; y < old_height; y++) {
237  for (unsigned int x = 0; x < old_width; x++) {
238  for (unsigned int value = 0; value < DISP_VALS; value++) {
239  imRef(data[i], x/2, y/2)[value] +=
240  imRef(data[i-1], x, y)[value];
241  }
242  }
243  }
244  }
245 
246  // run bp from coarse to fine
247  for (int i = alg_settings.num_levels - 1; i >= 0; i--) {
248  unsigned int width = (unsigned int)data[i]->width();
249  unsigned int height = (unsigned int)data[i]->height();
250 
251  // allocate & init memory for messages
252  if ((unsigned int)i == (alg_settings.num_levels - 1)) {
253  // in the coarsest level messages are initialized to zero
254  u[i] = new image<float[DISP_VALS]>(width, height);
255  d[i] = new image<float[DISP_VALS]>(width, height);
256  l[i] = new image<float[DISP_VALS]>(width, height);
257  r[i] = new image<float[DISP_VALS]>(width, height);
258  } else {
259  // initialize messages from values of previous level
260  u[i] = new image<float[DISP_VALS]>(width, height, false);
261  d[i] = new image<float[DISP_VALS]>(width, height, false);
262  l[i] = new image<float[DISP_VALS]>(width, height, false);
263  r[i] = new image<float[DISP_VALS]>(width, height, false);
264 
265  for (unsigned int y = 0; y < height; y++) {
266  for (unsigned int x = 0; x < width; x++) {
267  for (unsigned int value = 0; value < DISP_VALS; value++) {
268  imRef(u[i], x, y)[value] =
269  imRef(u[i+1], x/2, y/2)[value];
270  imRef(d[i], x, y)[value] =
271  imRef(d[i+1], x/2, y/2)[value];
272  imRef(l[i], x, y)[value] =
273  imRef(l[i+1], x/2, y/2)[value];
274  imRef(r[i], x, y)[value] =
275  imRef(r[i+1], x/2, y/2)[value];
276  }
277  }
278  }
279  // delete old messages and data
280  delete u[i + 1];
281  delete d[i + 1];
282  delete l[i + 1];
283  delete r[i + 1];
284  delete data[i + 1];
285  }
286 
287  // BP
288  bp_cb(u[i], d[i], l[i], r[i], data[i], alg_settings.num_iterations, alg_settings.disc_k_bp);
289  }
290 
291  image<uchar> *out = output(u[0], d[0], l[0], r[0], data[0]);
292 
293  auto timeEnd = std::chrono::system_clock::now();
294  runtime = timeEnd-timeStart;
295 
296  RunData run_data;
297  run_data.AddDataWHeader(std::string(run_eval::kSingleThreadRuntimeHeader), runtime.count());
298 
299  delete u[0];
300  delete d[0];
301  delete l[0];
302  delete r[0];
303  delete data[0];
304 
305  return {out, run_data};
306 }
