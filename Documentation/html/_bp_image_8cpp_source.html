31 void BpImage<T>::LoadImageAsGrayScale(const std::string& file_path_image) {
32  //check if PGM or PPM image (types currently supported)
33  std::istringstream iss(file_path_image);
34  std::string token;
35  while (std::getline(iss, token, '.')) {
36  //continue to get last token with the file extension
37  }
38 
39  //last token after "." is file extension
40  //use extension to check if image is pgm or ppm
41  BpImage<unsigned char> init_image;
42  if (token == beliefprop::kPGMExt) {
43  init_image = ImageRead(file_path_image, beliefprop::ImageType::kPgmImage);
44  } else if (token == beliefprop::kPPMExt) {
45  init_image = ImageRead(
46  file_path_image, beliefprop::ImageType::kPpmImage,
47  beliefprop::kUseWeightedRGBToGrayscaleConversion);
48  } else {
49  //end function if not of type pgm or ppm with result
50  //being BpImage with default constructor
51  return;
52  }
53 
54  //set width and height using data from file image and initialize
55  //pixels array of size width_ * height_ with unique pointer
56  width_height_ = {init_image.Width(), init_image.Height()};
57  pixels_ = std::make_unique<T[]>(TotalPixels());
58 
59  //convert each pixel in dataRead to data type T and place in imageData array in same location
60  std::ranges::transform(
61  init_image.PointerToPixelsStart(),
62  init_image.PointerToPixelsStart() + TotalPixels(),
63  pixels_.get(), [] (const unsigned char i) -> T {return (T)i;});
64 }

85 BpImage<unsigned char> BpImage<T>::ImageRead(const std::string& file_name,
86  beliefprop::ImageType image_type, bool weighted_rgb_conversion) const
87 {
88  std::string buf;
89 
90  /* read header */
91  std::ifstream file(file_name, std::ios::in | std::ios::binary);
92  pnm_read(file, buf);
93  if (buf != "P5")
94  std::cout << "ERROR READING FILE P5: " << file_name << std::endl;
95 
96  unsigned int cols, rows;
97  pnm_read(file, buf);
98  cols = std::stoul(buf);
99  pnm_read(file, buf);
100  rows = std::stoul(buf);
101 
102  pnm_read(file, buf);
103  if (std::stoul(buf) > UCHAR_MAX)
104  std::cout << "ERROR READING FILE UCHAR MAX: " << file_name << std::endl;
105 
106  BpImage<unsigned char> out_image(std::array<unsigned int, 2>{cols, rows});
107 
108  if (image_type == beliefprop::ImageType::kPgmImage) {
109  /* read data */
110  file.read((char*) (out_image.PointerToPixelsStart()),
111  (cols * rows * sizeof(char)));
112  }
113  else if (image_type == beliefprop::ImageType::kPpmImage) {
114  std::unique_ptr<char[]> rgb_image_ptr = std::make_unique<char[]>(3 * cols * rows);
115 
116  /* read data */
117  file.read(rgb_image_ptr.get(), 3 * cols * rows * sizeof(char));
118  file.close();
119 
120  //convert the RGB image to grayscale either
121  //with each channel weighted the same or weighted
122  //if weighted_rgb_conversion set to true
123  for (unsigned int i = 0; i < (rows * cols); i++) {
124  float r_channel_weight = 1.0f / 3.0f;
125  float b_channel_weight = 1.0f / 3.0f;
126  float g_channel_weight = 1.0f / 3.0f;
127  if (weighted_rgb_conversion) {
128  r_channel_weight = 0.299f;
129  b_channel_weight = 0.587f;
130  g_channel_weight = 0.114f;
131  }
132  out_image.PointerToPixelsStart()[i] = (unsigned char)std::floor(
133  r_channel_weight * ((float) rgb_image_ptr[i * 3]) +
134  g_channel_weight * ((float) rgb_image_ptr[i * 3 + 1]) +
135  b_channel_weight * ((float) rgb_image_ptr[i * 3 + 2]) +
136  0.5f);
137  }
138  }
139 
140  file.close();
141  return out_image;
142 }
