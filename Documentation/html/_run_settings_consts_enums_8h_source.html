66  if constexpr (ACCELERATION_SETTING == AccSetting::kNEON) { return "NEON"; }
67  else if constexpr (ACCELERATION_SETTING == AccSetting::kAVX256) { return "AVX256"; }
68  else if constexpr (ACCELERATION_SETTING == AccSetting::kAVX512) { return "AVX512"; }
