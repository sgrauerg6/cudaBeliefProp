42  inline RunData retrieveDeviceProperties(int num_device)
43  {
44  cudaDeviceProp prop;
45  std::array<int, 2> cuda_version_driver_runtime;
46  cudaGetDeviceProperties(&prop, num_device);
47  cudaDriverGetVersion(&(cuda_version_driver_runtime[0]));
48  cudaRuntimeGetVersion(&(cuda_version_driver_runtime[1]));
49 
50  RunData run_data;
51  run_data.AddDataWHeader("Device " + std::to_string(num_device),
52  std::string(prop.name) + " with " + std::to_string(prop.multiProcessorCount) +
53  " multiprocessors");
54  run_data.AddDataWHeader(std::string(beliefprop::kCUDAVersionHeader),
55  std::to_string(cuda_version_driver_runtime[0]));
56  run_data.AddDataWHeader(std::string(beliefprop::kCUDARuntimeHeader),
57  std::to_string(cuda_version_driver_runtime[1]));
58  return run_data;
59  }

69  //run the disparity map estimation BP on a set of stereo images and save the results between each set of images
70  std::optional<ProcessStereoSetOutput> operator()(const std::array<std::string, 2>& ref_test_image_path,
71  const beliefprop::BpSettings& alg_settings,
72  const ParallelParams& parallel_params) const override
73  {
74  //return no value if acceleration setting is not CUDA
75  if constexpr (ACCELERATION != run_environment::AccSetting::kCUDA) {
76  return {};
77  }
78 
79  //generate struct with pointers to objects for running CUDA implementation and call
80  //function to run CUDA implementation
81  RunData run_data;
82  run_data.AppendData(bp_cuda_device::retrieveDeviceProperties(0));
83  auto process_set_output = this->ProcessStereoSet(ref_test_image_path, alg_settings,
84  BpOnDevice<T, DISP_VALS, ACCELERATION>{
85  std::make_unique<SmoothImageCUDA>(parallel_params),
86  std::make_unique<ProcessCUDABP<T, DISP_VALS, ACCELERATION>>(parallel_params),
87  std::make_unique<RunImpCUDAMemoryManagement<T>>(),
88  std::make_unique<RunImpCUDAMemoryManagement<float>>()});
89  if (process_set_output) {
90  run_data.AppendData(process_set_output->run_data);
91  process_set_output->run_data = run_data;
92  }
93 
94  return process_set_output;
95  }
