37 //run implementation using input parameters from command line with specified setting
38 void runImp(int argc, char** argv, RunImpSetting impSetting)
39 {
40  //initialize settings to run implementation and evaluation
41  run_environment::RunImpSettings run_imp_settings;
42 
43  //enable optimization of parallel parameters with setting to use the same parallel parameters for all kernels in run
44  //testing on i7-11800H has found that using different parallel parameters (corresponding to OpenMP thread counts)
45  //in different kernels in the optimized CPU implementation can increase runtime (may want to test on additional processors)
46  run_imp_settings.opt_parallel_params_setting =
47  {true,
48  run_environment::OptParallelParamsSetting::kSameParallelParamsAllKernels};
49 
50  //set default parallel parameters and parallel parameters to benchmark when searching for optimal
51  //parallel parameters
52  run_imp_settings.p_params_default_opt_settings =
53  {run_cpu::kParallelParamsDefault,
54  run_cpu::kParallelParameterOptions};
55 
56  //set run name to first argument if it exists
57  //otherwise set to "CurrentRun"
58  run_imp_settings.run_name = (argc > 1) ? argv[1] : "CurrentRun";
59 
60  //adjust thread count to simulate single CPU on dual-CPU system
61  //currently only works as expected if environment variables are set before run such that threads are pinned to socket via
62  //"export OMP_PLACES="sockets"" and "export OMP_PROC_BIND=true" commands on command line
63  if (impSetting == RunImpSetting::kRunImpSimSingleCPUTwoCPUSystem)
64  {
65  //adjust settings to simulate run on single CPU in two-CPU system, specifically set parallel thread count options so that
66  //maximum number of parallel threads is thread count of single CPU and set environment variables so that CPU threads
67  //are pinned to socket
68  //set default parallel threads count to be number of threads on a single CPU in the two-CPU system
69  run_imp_settings.p_params_default_opt_settings.first =
70  {std::thread::hardware_concurrency() / 2, 1};
71 
72  //erase parallel thread count options with more than the number of threads on a single CPU in the two-CPU system
73  run_imp_settings.RemoveParallelParamAboveMaxThreads(
74  std::thread::hardware_concurrency() / 2);
75 
76  //adjust settings so that CPU threads pinned to socket to simulate run on single CPU
77  //TODO: commented out since currently has no effect; needs to be set before run by
78  //calling "export OMP_PLACES="sockets"" and "export OMP_PROC_BIND=true" commands on
79  //command line before run
80  //run_environment::CPUThreadsPinnedToSocket().operator()(true);
81 
82  //append run name to specify that simulating single CPU on dual-CPU system
83  if (run_imp_settings.run_name) {
84  *(run_imp_settings.run_name) += "_SimSingleCPUOnDualCPUSystem";
85  }
86  }
87  //check if running implementation with CPU threads pinned to socket (for cases with multiple CPUs)
88  //TODO: currently not supported due to code for setting CPU threads to be pinned to socket not working as expected
89  //can still run implementation with CPU threads pinned to socket by calling "export OMP_PLACES="sockets"" and
90  //"export OMP_PROC_BIND=true" commands on command line before run
91  /*else if (impSetting == RunImpSetting::kRunImpThreadsPinnedToSocket) {
92  //adjust settings so that CPU threads pinned to socket
93  //TODO: commented out since currently has no effect
94  //run_environment::CPUThreadsPinnedToSocket().operator()(true);
95 
96  //append run name to specify that CPU threads pinned to socket
97  if (run_imp_settings.run_name) {
98  *(run_imp_settings.run_name) += "_ThreadsPinnedToSocket";
99  }
100  }*/
101 
102  //set datatype(s) to use in run processing in evaluation
103  run_imp_settings.datatypes_eval_sizes =
104  {run_eval::kDataTypesEvalSizes.begin(),
105  run_eval::kDataTypesEvalSizes.end()};
106 
107  //remove any parallel processing below given minimum number of threads
108  run_imp_settings.RemoveParallelParamBelowMinThreads(
109  run_cpu::kMinNumThreadsRun);
110 
111  //set setting of whether or not to use templated loop iterations in implementation
112  //in evaluation runs
113  run_imp_settings.templated_iters_setting =
114  run_environment::TemplatedItersSetting::kRunOnlyNonTemplated;
115 
116  //set path of baseline runtimes and baseline description
117  run_imp_settings.baseline_runtimes_path_desc =
118  {bp_file_handling::kBaselineRunDataPath,
119  bp_file_handling::kBaselineRunDesc};
120 
121  //set data subsets to evaluate separate from all data
122  run_imp_settings.subset_desc_input_sig = beliefprop::kEvalDataSubsets;
123 
124  //run belief propagation with AVX512, AVX256, and no vectorization implementations,
125  //with the AVX512 implementation given first as the expected fastest implementation
126  RunImpMultTypesAccels().operator()({
127  {run_environment::AccSetting::kAVX512,
128  std::make_shared<RunBpImpMultInputs>(run_environment::AccSetting::kAVX512)},
129  {run_environment::AccSetting::kAVX256,
130  std::make_shared<RunBpImpMultInputs>(run_environment::AccSetting::kAVX256)},
131  {run_environment::AccSetting::kNone,
132  std::make_shared<RunBpImpMultInputs>(run_environment::AccSetting::kNone)}},
133  run_imp_settings,
134  std::make_unique<EvaluateBpImpResults>());
135 }
136 
137 int main(int argc, char** argv)
138 {
139  //if running on a system with two cpus, run implementation with settings adjusted
140  //to simulate run on single CPU if specified in second command line argument
141  if ((argc > 2) && (std::string(argv[2]) == std::string(run_cpu::kSimulateSingleCPU))) {
142  std::cout << "Running optimized CPU implementation with settings adjusted such that "
143  "a single CPU is simulated on a dual-CPU system." << std::endl;
144  std::cout << "Results only as expected if running on dual-CPU system and "
145  "environment variables set so that threads pinned to CPU socket." << std::endl;
146  runImp(argc, argv, RunImpSetting::kRunImpSimSingleCPUTwoCPUSystem);
147  }
148  else {
149  //run default implementation
150  std::cout << "Running optimized CPU implementation" << std::endl;
151  runImp(argc, argv, RunImpSetting::kRunImpDefault);
152  }
153 
154  return 0;
