32 std::string RunData::GetHeaderToAdd(const std::string& in_header) const {
33  auto headerToAdd = in_header;
34  unsigned int num{0};
35  while (std::find(headers_in_order_.cbegin(), headers_in_order_.cend(), headerToAdd) != headers_in_order_.cend()) {
36  //add "_{num}" to header if header already in data
37  num++;
38  headerToAdd = in_header + "_" + std::to_string(num);
39  }
40  return headerToAdd;
41 }

79 void RunData::AppendData(const RunData& rundata) {
80  //const auto& rundataMapping = rundata.GetAllData();
81  for (const auto& header : rundata.HeadersInOrder()) {
82  if (rundata.GetDataAsDouble(header)) {
83  AddDataWHeader(header, *(rundata.GetDataAsDouble(header)));
84  }
85  else if (rundata.GetDataAsUInt(header)) {
86  AddDataWHeader(header, *(rundata.GetDataAsUInt(header)));
87  }
88  else if (rundata.GetDataAsBool(header)) {
89  AddDataWHeader(header, *(rundata.GetDataAsBool(header)));
90  }
91  else {
92  AddDataWHeader(header, rundata.GetDataAsStr(header));
93  }
94  }

98 std::string RunData::GetDataAsStr(const std::string_view header) const {
99  const auto variantVal = headers_w_data_.at(std::string(header));
100  if (std::holds_alternative<std::string>(variantVal)) {
101  return std::get<std::string>(variantVal);
102  }
103  if (std::holds_alternative<double>(variantVal)) {
104  return std::to_string(std::get<double>(variantVal));
105  }
106  if (std::holds_alternative<unsigned int>(variantVal)) {
107  return std::to_string(std::get<unsigned int>(variantVal));
108  }
109  if (std::holds_alternative<bool>(variantVal)) {
110  return (std::get<bool>(variantVal)) ?
111  std::string(run_eval::kBoolValFalseTrueDispStr[1]) :
112  std::string(run_eval::kBoolValFalseTrueDispStr[0]);
113  }
114  return "";
115 }
