168 //run and evaluate optimized belief propagation implementation on evaluation stereo set
169 //specified by NUM_INPUT
170 //data type used in implementation specified by T
171 //bp implemenation optimization specified by OPT_IMP_ACCEL
172 //evaluation stereo set to run implementation on specified by NUM_INPUT
173 template<RunData_t T, run_environment::AccSetting OPT_IMP_ACCEL, unsigned int NUM_INPUT>
174 MultRunData RunImpOnInputBp<T, OPT_IMP_ACCEL, NUM_INPUT>::operator()(
175  const run_environment::RunImpSettings& run_imp_settings)
176 {
177  //set up BP settings for current run
178  alg_settings_.num_disp_vals = beliefprop::kStereoSetsToProcess[NUM_INPUT].num_disp_vals;
179  alg_settings_.disc_k_bp = (float)alg_settings_.num_disp_vals / 7.5f;
180 
181  //initialize run results across multiple implementations
182  MultRunData run_results;
183 
184  //set up unoptimized single threaded bp stereo implementation
185  run_bp_stereo_single_thread_ =
186  std::make_unique<RunBpOnStereoSetSingleThreadCPU<T, beliefprop::kStereoSetsToProcess[NUM_INPUT].num_disp_vals, run_environment::AccSetting::kNone>>();
187 
188  //set up and run bp stereo using optimized implementation (optimized CPU and CUDA implementations supported)
189  //as well as unoptimized implementation for comparison
190  //run optimized implementation with and/or without disparity value count templated depending on setting
191  //for running implementation with disparity values templated
192  if (run_imp_settings.templated_iters_setting != run_environment::TemplatedItersSetting::kRunOnlyNonTemplated) {
193  run_opt_bp_num_iters_templated_ =
194  std::make_unique<RunBpOptimized<T, beliefprop::kStereoSetsToProcess[NUM_INPUT].num_disp_vals, OPT_IMP_ACCEL>>();
195  constexpr bool run_w_loop_iters_templated{true};
196  InputSignature input_sig(sizeof(T), NUM_INPUT, run_w_loop_iters_templated);
197  run_results.insert({input_sig, (this->RunEvalBenchmark(run_imp_settings, run_w_loop_iters_templated))});
198  }
199  if (run_imp_settings.templated_iters_setting != run_environment::TemplatedItersSetting::kRunOnlyTempated) {
200  run_opt_bp_num_iters_no_template_ =
201  std::make_unique<RunBpOptimized<T, 0, OPT_IMP_ACCEL>>();
202  constexpr bool run_w_loop_iters_templated{false};
203  InputSignature input_sig(sizeof(T), NUM_INPUT, run_w_loop_iters_templated);
204  run_results.insert({input_sig, this->RunEvalBenchmark(run_imp_settings, run_w_loop_iters_templated)});
205  }
206 
207  //return bp run results across multiple implementations
208  return run_results;
209 }

237 //run and compare output disparity maps using the given optimized and single-threaded stereo implementations
238 //on the reference and test images specified by numStereoSet
239 //run only optimized implementation if run_opt_imp_only is true
240 template<RunData_t T, run_environment::AccSetting OPT_IMP_ACCEL, unsigned int NUM_INPUT>
241 std::optional<RunData> RunImpOnInputBp<T, OPT_IMP_ACCEL, NUM_INPUT>::RunImpsAndCompare(
242  std::shared_ptr<ParallelParams> parallel_params,
243  bool run_opt_imp_only,
244  bool run_imp_templated_loop_iters) const
245 {
246  //get properties of input stereo set from stereo set number
247  BpFileHandling bp_file_settings(
248  std::string(beliefprop::kStereoSetsToProcess[NUM_INPUT].name));
249  const std::array<filepathtype, 2> ref_test_image_path{
250  bp_file_settings.RefImagePath(), bp_file_settings.TestImagePath()};
251 
252  //get number of implementations to run output disparity map file path(s)
253  //if run_opt_imp_only is false, run single-threaded implementation in
254  //addition to optimized implementation
255  const unsigned int num_imps_run{run_opt_imp_only ? 1u : 2u};
256  std::array<filepathtype, 2> output_disp;
257  for (unsigned int i=0; i < num_imps_run; i++) {
258  output_disp[i] = bp_file_settings.GetCurrentOutputDisparityFilePathAndIncrement();
259  }
260 
261  //get optimized implementation description and write info about run to std::cout stream
262  const std::string opt_imp_run_description{run_imp_templated_loop_iters ?
263  run_opt_bp_num_iters_templated_->BpRunDescription() :
264  run_opt_bp_num_iters_no_template_->BpRunDescription()};
265  std::cout << "Running belief propagation on reference image "
266  << ref_test_image_path[0] << " and test image "
267  << ref_test_image_path[1] << " on " << opt_imp_run_description;
268  if (!run_opt_imp_only) {
269  std::cout << " and " << run_bp_stereo_single_thread_->BpRunDescription();
270  }
271  std::cout << std::endl;
272 
273  //run optimized implementation and retrieve structure with runtime and output disparity map
274  std::map<run_environment::AccSetting, std::optional<beliefprop::BpRunOutput>> run_output;
275  if (run_imp_templated_loop_iters) {
276  run_output[OPT_IMP_ACCEL] = run_opt_bp_num_iters_templated_->operator()(
277  {ref_test_image_path[0].string(), ref_test_image_path[1].string()},
278  alg_settings_,
279  *parallel_params);
280  }
281  else {
282  run_output[OPT_IMP_ACCEL] = run_opt_bp_num_iters_no_template_->operator()(
283  {ref_test_image_path[0].string(), ref_test_image_path[1].string()},
284  alg_settings_,
285  *parallel_params);
286  }
287 
288  //check if error in run
289  RunData run_data;
290  run_data.AddDataWHeader(
291  std::string(run_environment::kAccelerationDescHeader), opt_imp_run_description);
292  if (!(run_output[OPT_IMP_ACCEL])) {
293  return {};
294  }
295 
296  //append implementation run data to evaluation run data
297  run_data.AppendData(run_output[OPT_IMP_ACCEL]->run_data);
298  run_data.AddDataWHeader(
299  std::string(run_eval::kOptimizedRuntimeHeader), run_output[OPT_IMP_ACCEL]->run_time.count());
300 
301  //save resulting disparity map
302  run_output[OPT_IMP_ACCEL]->out_disparity_map.SaveDisparityMap(
303  output_disp[0].string(),
304  beliefprop::kStereoSetsToProcess[NUM_INPUT].scale_factor);
305 
306  //check if only running optimized implementation or if also running single-threaded implementation
307  if (!run_opt_imp_only) {
308  //check if running single threaded implementation or using stored results from previous run
309  if ((!(bp_single_thread::kRunSingleThreadOnceForSet)) ||
310  (!(bp_single_thread::single_thread_run_output.contains(ref_test_image_path))))
311  {
312  //run single-threaded implementation and retrieve structure with runtime and output disparity map
313  run_output[run_environment::AccSetting::kNone] = run_bp_stereo_single_thread_->operator()(
314  {ref_test_image_path[0].string(), ref_test_image_path[1].string()},
315  alg_settings_, *parallel_params);
316  if (!(run_output[run_environment::AccSetting::kNone])) {
317  return {};
318  }
319 
320  //save resulting disparity map
321  run_output[run_environment::AccSetting::kNone]->out_disparity_map.SaveDisparityMap(
322  output_disp[1].string(),
323  beliefprop::kStereoSetsToProcess[NUM_INPUT].scale_factor);
324 
325  //save run data and result if setting to only run single thread implementation once for each set
326  if (bp_single_thread::kRunSingleThreadOnceForSet) {
327  bp_single_thread::single_thread_run_output[ref_test_image_path] =
328  {run_output[run_environment::AccSetting::kNone]->run_time,
329  run_output[run_environment::AccSetting::kNone]->run_data,
330  output_disp[1]};
331  }
332  }
333  else {
334  run_output[run_environment::AccSetting::kNone] = beliefprop::BpRunOutput();
335  //retrieve stored results from previous run on single threaded implementation
336  run_output[run_environment::AccSetting::kNone]->run_time =
337  std::get<0>(bp_single_thread::single_thread_run_output[ref_test_image_path]);
338  run_output[run_environment::AccSetting::kNone]->run_data =
339  std::get<1>(bp_single_thread::single_thread_run_output[ref_test_image_path]);
340  run_output[run_environment::AccSetting::kNone]->out_disparity_map =
341  DisparityMap<float>(
342  std::get<2>(bp_single_thread::single_thread_run_output[ref_test_image_path]).string(),
343  beliefprop::kStereoSetsToProcess[NUM_INPUT].scale_factor);
344  }
345  //append run data for single thread run to evaluation run data
346  run_data.AppendData(run_output[run_environment::AccSetting::kNone]->run_data);
347  }
348 
349  //write location of output disparity maps for each implementation to std::cout
350  for (unsigned int i = 0; i < num_imps_run; i++) {
351  const std::string run_description{(i == 0) ?
352  opt_imp_run_description :
353  run_bp_stereo_single_thread_->BpRunDescription()};
354  std::cout << "Output disparity map from " << run_description
355  << " run at " << output_disp[i] << std::endl;
356  }
357  std::cout << std::endl;
358 
359  //compare resulting disparity maps with ground truth and to each other
360  const filepathtype ground_truth_disp{bp_file_settings.GroundTruthDisparityFilePath()};
361  const DisparityMap<float> ground_truth_disparity_map(
362  ground_truth_disp.string(),
363  beliefprop::kStereoSetsToProcess[NUM_INPUT].scale_factor);
364  run_data.AddDataWHeader(opt_imp_run_description + " output vs. Ground Truth result", std::string());
365  run_data.AppendData(run_output[OPT_IMP_ACCEL]->out_disparity_map.OutputComparison(
366  ground_truth_disparity_map, beliefprop::DisparityMapEvaluationParams()).AsRunData());
367  if (!run_opt_imp_only) {
368  run_data.AddDataWHeader(
369  run_bp_stereo_single_thread_->BpRunDescription() + " output vs. Ground Truth result", std::string());
370  run_data.AppendData(
371  run_output[run_environment::AccSetting::kNone]->out_disparity_map.OutputComparison(
372  ground_truth_disparity_map, beliefprop::DisparityMapEvaluationParams()).AsRunData());
373  run_data.AddDataWHeader(
374  opt_imp_run_description + " output vs. " + run_bp_stereo_single_thread_->BpRunDescription() +
375  " result", std::string());
376  run_data.AppendData(run_output[OPT_IMP_ACCEL]->out_disparity_map.OutputComparison(
377  run_output[run_environment::AccSetting::kNone]->out_disparity_map,
378  beliefprop::DisparityMapEvaluationParams()).AsRunData());
379  }
380 
381  //return structure indicating that run succeeded along with data from run
382  return run_data;
383 }
