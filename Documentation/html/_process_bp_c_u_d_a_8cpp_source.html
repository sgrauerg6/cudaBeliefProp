34 //return whether or not there was an error in CUDA processing
35 template<RunData_t T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
36 inline run_eval::Status ProcessBpCUDA<T, DISP_VALS, ACCELERATION>::ErrorCheck(const char *file, int line, bool abort) const {
37  const auto code = cudaPeekAtLastError();
38  if (code != cudaSuccess) {
39  std::cout << "CUDA ERROR: " << cudaGetErrorString(code) << " " << file << " " << line << std::endl;
40  cudaGetLastError();
41  cudaDeviceReset();
42  cudaDeviceSynchronize();
43  cudaSetDevice(0);
44  if (abort) {
45  exit(code);
46  }
47  return run_eval::Status::kError;
48  }
49  return run_eval::Status::kNoError;
50 }

54 //run the given number of iterations of BP at the current level using the given message values in global device memory
55 template<RunData_t T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
56 run_eval::Status ProcessBpCUDA<T, DISP_VALS, ACCELERATION>::RunBPAtCurrentLevel(
57  const beliefprop::BpSettings& alg_settings,
58  const BpLevel& current_bp_level,
59  const beliefprop::DataCostsCheckerboards<T*>& data_costs_device,
60  const beliefprop::CheckerboardMessages<T*>& messages_device,
61  T* allocated_memory) const
62 {
63  //set to prefer L1 cache since shared memory is not used in this implementation
64  cudaDeviceSetCacheConfig(cudaFuncCachePreferL1);
65  const auto kernel_thread_block_dims = this->parallel_params_.OptParamsForKernel(
66  {static_cast<unsigned int>(beliefprop::BpKernel::kBpAtLevel), current_bp_level.LevelProperties().level_num_});
67  const dim3 threads{kernel_thread_block_dims[0], kernel_thread_block_dims[1]};
68  //only updating half at a time
69  const dim3 grid{(unsigned int)ceil((float)(current_bp_level.LevelProperties().width_checkerboard_level_) / (float)threads.x),
70  (unsigned int)ceil((float)current_bp_level.LevelProperties().height_level_ / (float)threads.y)};
71 
72  //in cuda kernel storing data one at a time (though it is coalesced), so simd_data_size not relevant here and set to 1
73  //still is a check if start of row is aligned
74  const bool data_aligned{beliefprop::MemoryAlignedAtDataStart(0, 1, current_bp_level.LevelProperties().num_data_align_width_,
75  current_bp_level.LevelProperties().div_padded_checkerboard_w_align_)};
76 
77  //at each level, run BP for numIterations, alternating between updating the messages between the two "checkerboards"
78  for (unsigned int iteration_num = 0; iteration_num < alg_settings.num_iterations; iteration_num++)
79  {
80  beliefprop::CheckerboardPart checkerboard_part_update =
81  ((iteration_num % 2) == 0) ?
82  beliefprop::CheckerboardPart::kCheckerboardPart1 :
83  beliefprop::CheckerboardPart::kCheckerboardPart0;
84  cudaDeviceSynchronize();
85 
86  using namespace beliefprop;
87  if constexpr (DISP_VALS > 0) {
88  beliefprop_cuda::RunBPIterationUsingCheckerboardUpdates<T, DISP_VALS> <<<grid, threads>>> (
89  checkerboard_part_update, current_bp_level.LevelProperties(),
90  data_costs_device[0], data_costs_device[1],
91  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesUCheckerboard)],
92  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesDCheckerboard)],
93  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesLCheckerboard)],
94  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesRCheckerboard)],
95  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesUCheckerboard)],
96  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesDCheckerboard)],
97  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesLCheckerboard)],
98  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesRCheckerboard)],
99  alg_settings.disc_k_bp, data_aligned, alg_settings.num_disp_vals);
100  }
101  else {
102  beliefprop_cuda::RunBPIterationUsingCheckerboardUpdates<T, DISP_VALS> <<<grid, threads>>> (
103  checkerboard_part_update, current_bp_level.LevelProperties(),
104  data_costs_device[0], data_costs_device[1],
105  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesUCheckerboard)],
106  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesDCheckerboard)],
107  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesLCheckerboard)],
108  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesRCheckerboard)],
109  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesUCheckerboard)],
110  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesDCheckerboard)],
111  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesLCheckerboard)],
112  messages_device[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesRCheckerboard)],
113  alg_settings.disc_k_bp, data_aligned, alg_settings.num_disp_vals, allocated_memory);
114  }
115 
116  cudaDeviceSynchronize();
117  if (ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
118  return run_eval::Status::kError;
119  }
120  }
121  return run_eval::Status::kNoError;
122 }

125 //pyramid; the next level down is double the width and height of the current level so each message in the current level is copied into four "slots"
126 //in the next level down
127 //need two different "sets" of message values to avoid read-write conflicts
128 template<RunData_t T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
129 run_eval::Status ProcessBpCUDA<T, DISP_VALS, ACCELERATION>::CopyMessageValuesToNextLevelDown(
130  const BpLevel& current_bp_level,
131  const BpLevel& next_bp_level,
132  const beliefprop::CheckerboardMessages<T*>& messages_device_copy_from,
133  const beliefprop::CheckerboardMessages<T*>& messages_device_copy_to,
134  unsigned int bp_settings_num_disp_vals) const
135 {
136  const auto kernel_thread_block_dims = this->parallel_params_.OptParamsForKernel(
137  {static_cast<unsigned int>(beliefprop::BpKernel::kCopyAtLevel), current_bp_level.LevelProperties().level_num_});
138  const dim3 threads{kernel_thread_block_dims[0], kernel_thread_block_dims[1]};
139  const dim3 grid{(unsigned int)ceil((float)(current_bp_level.LevelProperties().width_checkerboard_level_) / (float)threads.x),
140  (unsigned int)ceil((float)(current_bp_level.LevelProperties().height_level_) / (float)threads.y)};
141 
142  cudaDeviceSynchronize();
143  if (ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
144  return run_eval::Status::kError;
145  }
146 
147  for (const auto checkerboard_part : {beliefprop::CheckerboardPart::kCheckerboardPart0,
148  beliefprop::CheckerboardPart::kCheckerboardPart1})
149  {
150  using namespace beliefprop;
151  //call the kernel to copy the computed BP message data to the next level down in parallel in each of the two "checkerboards"
152  //storing the current message values
153  beliefprop_cuda::CopyMsgDataToNextLevel<T, DISP_VALS> <<< grid, threads >>> (
154  checkerboard_part, current_bp_level.LevelProperties(), next_bp_level.LevelProperties(),
155  messages_device_copy_from[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesUCheckerboard)],
156  messages_device_copy_from[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesDCheckerboard)],
157  messages_device_copy_from[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesLCheckerboard)],
158  messages_device_copy_from[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesRCheckerboard)],
159  messages_device_copy_from[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesUCheckerboard)],
160  messages_device_copy_from[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesDCheckerboard)],
161  messages_device_copy_from[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesLCheckerboard)],
162  messages_device_copy_from[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesRCheckerboard)],
163  messages_device_copy_to[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesUCheckerboard)],
164  messages_device_copy_to[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesDCheckerboard)],
165  messages_device_copy_to[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesLCheckerboard)],
166  messages_device_copy_to[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart0)][static_cast<unsigned int>(MessageArrays::kMessagesRCheckerboard)],
167  messages_device_copy_to[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesUCheckerboard)],
168  messages_device_copy_to[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesDCheckerboard)],
169  messages_device_copy_to[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesLCheckerboard)],
170  messages_device_copy_to[static_cast<unsigned int>(CheckerboardPart::kCheckerboardPart1)][static_cast<unsigned int>(MessageArrays::kMessagesRCheckerboard)],
171  bp_settings_num_disp_vals);
172 
173  cudaDeviceSynchronize();
174  if (ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
175  return run_eval::Status::kError;
176  }
177  }
178  return run_eval::Status::kNoError;

181 //initialize the data cost at each pixel with no estimated Stereo values...only the data and discontinuity costs are used
182 template<RunData_t T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
183 run_eval::Status ProcessBpCUDA<T, DISP_VALS, ACCELERATION>::InitializeDataCosts(
184  const beliefprop::BpSettings& alg_settings,
185  const BpLevel& current_bp_level,
186  const std::array<float*, 2>& images_target_device,
187  const beliefprop::DataCostsCheckerboards<T*>& data_costs_device) const
188 {
189  if (ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
190  return run_eval::Status::kError;
191  }
192 
193  //since this is first kernel run in BP, set to prefer L1 cache for now since no shared memory is used by default
194  cudaDeviceSetCacheConfig(cudaFuncCachePreferL1);
195 
196  //setup execution parameters
197  //the thread size remains constant throughout but the grid size is adjusted based on the current level/kernel to run
198  const auto kernel_thread_block_dims = this->parallel_params_.OptParamsForKernel(
199  {static_cast<unsigned int>(beliefprop::BpKernel::kDataCostsAtLevel), 0});
200  const dim3 threads{kernel_thread_block_dims[0], kernel_thread_block_dims[1]};
201  //kernel run on full-sized image to retrieve data costs at the "bottom" level of the pyramid
202  const dim3 grid{(unsigned int)ceil((float)current_bp_level.LevelProperties().width_level_ / (float)threads.x),
203  (unsigned int)ceil((float)current_bp_level.LevelProperties().height_level_ / (float)threads.y)};
204 
205  //initialize the data the the "bottom" of the image pyramid
206  beliefprop_cuda::InitializeBottomLevelData<T, DISP_VALS> <<<grid, threads>>> (
207  current_bp_level.LevelProperties(),
208  images_target_device[0], images_target_device[1],
209  data_costs_device[0], data_costs_device[1],
210  alg_settings.lambda_bp, alg_settings.data_k_bp, alg_settings.num_disp_vals);
211  cudaDeviceSynchronize();
212 
213  if (ErrorCheck(__FILE__, __LINE__) != run_eval::Status::kNoError) {
214  return run_eval::Status::kError;
215  }
216 
217  return run_eval::Status::kNoError;
218 }

255 template<RunData_t T, unsigned int DISP_VALS, run_environment::AccSetting ACCELERATION>
256 run_eval::Status ProcessBpCUDA<T, DISP_VALS, ACCELERATION>::InitializeDataCurrentLevel(
257  const BpLevel& current_bp_level, const BpLevel& prev_bp_level,
258  const beliefprop::DataCostsCheckerboards<T*>& data_costs_device,
259  const beliefprop::DataCostsCheckerboards<T*>& data_costs_device_write,
260  unsigned int bp_settings_num_disp_vals) const
261 {
262  const auto kernel_thread_block_dims = this->parallel_params_.OptParamsForKernel(
263  {static_cast<unsigned int>(beliefprop::BpKernel::kDataCostsAtLevel), current_bp_level.LevelProperties().level_num_});
264  const dim3 threads{kernel_thread_block_dims[0], kernel_thread_block_dims[1]};
265  //each pixel "checkerboard" is half the width of the level and there are two of them
266  //each "pixel/point" at the level belongs to one checkerboard and
267  //the four-connected neighbors of the pixel are in the counterpart checkerboard
268  const dim3 grid{(unsigned int)ceil(((float)current_bp_level.LevelProperties().width_checkerboard_level_) / (float)threads.x),
269  (unsigned int)ceil((float)current_bp_level.LevelProperties().height_level_ / (float)threads.y)};
270 
271  if (ErrorCheck(__FILE__, __LINE__ ) != run_eval::Status::kNoError) {
272  return run_eval::Status::kError;
273  }
274 
275  const size_t offset_num{0};
276  for (const auto& checkerboard_data_cost : {
277  std::make_pair(beliefprop::CheckerboardPart::kCheckerboardPart0, data_costs_device_write[0]),
278  std::make_pair(beliefprop::CheckerboardPart::kCheckerboardPart1, data_costs_device_write[1])})
279  {
280  beliefprop_cuda::InitializeCurrentLevelData<T, DISP_VALS> <<<grid, threads>>> (
281  checkerboard_data_cost.first,
282  current_bp_level.LevelProperties(), prev_bp_level.LevelProperties(),
283  data_costs_device[0], data_costs_device[1],
284  checkerboard_data_cost.second, ((unsigned int) offset_num / sizeof(float)),
285  bp_settings_num_disp_vals);
286 
287  cudaDeviceSynchronize();
288  if (ErrorCheck(__FILE__, __LINE__ ) != run_eval::Status::kNoError) {
289  return run_eval::Status::kError;
290  }
291  }
292  return run_eval::Status::kNoError;
293 }
