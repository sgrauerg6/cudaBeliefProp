118 void DetailedTimings<T>::AddToCurrentTimings(
119  const DetailedTimings& in_detailed_timings)
120 {
121  std::ranges::for_each(in_detailed_timings.segment_timings_,
122  [this](const auto& current_timing) {
123  auto iter = this->segment_timings_.find(current_timing.first);
124  if (iter != this->segment_timings_.cend()) {
125  iter->second.insert(iter->second.cend(),
126  current_timing.second.cbegin(),
127  current_timing.second.cend());
128  }
129  else {
130  this->segment_timings_[current_timing.first] = current_timing.second;
131  }
132  });
133 }

138 std::chrono::duration<double> DetailedTimings<T>::MedianTiming(
139  const T run_segment_index) const
140 {
141  if (segment_timings_.at(run_segment_index).size() > 0) {
142  std::vector<std::chrono::duration<double>> segment_timing_vect_copy(
143  segment_timings_.at(run_segment_index));
144  //get median timing across runs
145  std::ranges::nth_element(
146  segment_timing_vect_copy,
147  segment_timing_vect_copy.begin() + segment_timing_vect_copy.size() / 2);
148  return (segment_timing_vect_copy[segment_timing_vect_copy.size() / 2]);
149  }
150  else {
151  return std::chrono::duration<double>();
152  }
