35 //evaluate results for implementation runs on multiple inputs with all the runs having the same data type and acceleration method
36 //return run data with speedup from evaluation of implementation runs using multiple inputs with runs
37 //having the same data type and acceleration method
38 std::pair<MultRunData, std::vector<RunSpeedupAvgMedian>> EvaluateImpResults::EvalResultsSingDataTypeAcc(
39  const MultRunData& run_results,
40  const run_environment::RunImpSettings run_imp_settings,
41  size_t data_size) const
42 {
43  //initialize and add speedup results over baseline data if available for current input
44  auto run_imp_opt_results = run_results;
45  const auto speedup_over_baseline =
46  GetSpeedupOverBaseline(run_imp_settings, run_imp_opt_results, data_size);
47  const auto speedup_over_baseline_subsets =
48  GetSpeedupOverBaselineSubsets(run_imp_settings, run_imp_opt_results, data_size);
49 
50  //initialize implementation run speedups
51  std::vector<RunSpeedupAvgMedian> run_imp_speedups;
52  run_imp_speedups.insert(run_imp_speedups.cend(),
53  speedup_over_baseline.cbegin(),
54  speedup_over_baseline.cend());
55  run_imp_speedups.insert(run_imp_speedups.cend(),
56  speedup_over_baseline_subsets.cbegin(),
57  speedup_over_baseline_subsets.cend());
58 
59  //compute and add speedup info for using optimized parallel parameters
60  //compared to default parallel parameters
61  if (run_imp_settings.opt_parallel_params_setting.first) {
62  const std::string speedup_header_optimized_p_params =
63  std::string(run_eval::kSpeedupOptParParamsHeader) + " - " +
64  std::string(run_environment::kDataSizeToNameMap.at(data_size));
65  run_imp_speedups.push_back(GetAvgMedSpeedupOptPParams(
66  run_imp_opt_results, speedup_header_optimized_p_params));
67  }
68 
69  //compute and add speedup info for using templated loop iteration counts compared to
70  //loop iteration counts not being known at compile time
71  if (run_imp_settings.templated_iters_setting == run_environment::TemplatedItersSetting::kRunTemplatedAndNotTemplated) {
72  const std::string speedup_header_loop_iters_templated =
73  std::string(run_eval::kSpeedupLoopItersCountTemplate) + " - " +
74  std::string(run_environment::kDataSizeToNameMap.at(data_size));
75  run_imp_speedups.push_back(GetAvgMedSpeedupLoopItersInTemplate(
76  run_imp_opt_results, speedup_header_loop_iters_templated));
77  }
78 
79  //return run data with speedup from evaluation of implementation runs using multiple inputs with runs
80  //having the same data type and acceleration method
81  return {run_imp_opt_results, run_imp_speedups};
82 }

84 //evaluate results for all implementation runs on multiple inputs with the runs
85 //potentially having different data types and acceleration methods and
86 //write run result and speedup outputs to files
87 void EvaluateImpResults::EvalAllResultsWriteOutput(
88  const std::unordered_map<size_t, MultRunDataWSpeedupByAcc>& run_results_mult_runs,
89  const run_environment::RunImpSettings run_imp_settings,
90  run_environment::AccSetting opt_imp_acc) const
91 {
92  std::unordered_map<size_t, MultRunDataWSpeedupByAcc> run_result_mult_runs_opt =
93  run_results_mult_runs;
94  //get speedup/slowdown using alternate accelerations
95  //and update optimized run result for input with result using alternate acceleration if it is faster than
96  //result with "optimal" acceleration for specific input
97  std::unordered_map<size_t, std::vector<RunSpeedupAvgMedian>> alt_imp_speedup;
98  std::unordered_map<size_t, RunSpeedupAvgMedian> alt_datatype_speedup;
99  for (const size_t data_size : run_imp_settings.datatypes_eval_sizes) {
100  alt_imp_speedup[data_size] = GetAltAccelSpeedups(
101  run_result_mult_runs_opt[data_size], run_imp_settings, data_size, opt_imp_acc);
102  if (data_size != sizeof(float)) {
103  //get speedup or slowdown using alternate data type (double or half) compared with float
104  alt_datatype_speedup[data_size] = GetAvgMedSpeedupBaseVsTarget(
105  run_result_mult_runs_opt[sizeof(float)][opt_imp_acc].first,
106  run_result_mult_runs_opt[data_size][opt_imp_acc].first,
107  (data_size > sizeof(float)) ? run_eval::kSpeedupDouble : run_eval::kSpeedupHalf,
108  BaseTargetDiff::kDiffDatatype);
109  }
110  }
111 
112  //initialize overall results to float results using fastest acceleration and add double and half-type results to it
113  auto results_w_speedups = run_result_mult_runs_opt[sizeof(float)][opt_imp_acc];
114  if (run_result_mult_runs_opt.contains(sizeof(double))) {
115  results_w_speedups.first.merge(
116  run_result_mult_runs_opt[sizeof(double)][opt_imp_acc].first);
117  }
118  if (run_result_mult_runs_opt.contains(sizeof(halftype))) {
119  results_w_speedups.first.merge(
120  run_result_mult_runs_opt[sizeof(halftype)][opt_imp_acc].first);
121  }
122 
123  //add speedup data from double and half precision runs to speedup results
124  results_w_speedups.second.insert(results_w_speedups.second.cend(),
125  alt_imp_speedup[sizeof(float)].cbegin(),
126  alt_imp_speedup[sizeof(float)].cend());
127  if (run_result_mult_runs_opt.contains(sizeof(double))) {
128  results_w_speedups.second.insert(results_w_speedups.second.cend(),
129  run_result_mult_runs_opt[sizeof(double)][opt_imp_acc].second.cbegin(),
130  run_result_mult_runs_opt[sizeof(double)][opt_imp_acc].second.cend());
131  results_w_speedups.second.insert(results_w_speedups.second.cend(),
132  alt_imp_speedup[sizeof(double)].cbegin(),
133  alt_imp_speedup[sizeof(double)].cend());
134  }
135  if (run_result_mult_runs_opt.contains(sizeof(halftype))) {
136  results_w_speedups.second.insert(results_w_speedups.second.cend(),
137  run_result_mult_runs_opt[sizeof(halftype)][opt_imp_acc].second.cbegin(),
138  run_result_mult_runs_opt[sizeof(halftype)][opt_imp_acc].second.cend());
139  results_w_speedups.second.insert(results_w_speedups.second.cend(),
140  alt_imp_speedup[sizeof(halftype)].cbegin(),
141  alt_imp_speedup[sizeof(halftype)].cend());
142  }
143 
144  //get speedup over baseline runtimes
145  if (run_imp_settings.baseline_runtimes_path_desc)
146  {
147  const auto speedup_over_baseline = GetAvgMedSpeedupOverBaseline(
148  results_w_speedups.first, run_eval::kAllRunsStr,
149  run_imp_settings.baseline_runtimes_path_desc.value());
150  results_w_speedups.second.insert(
151  results_w_speedups.second.cend(),
152  speedup_over_baseline.cbegin(),
153  speedup_over_baseline.cend());
154  }
155 
156  //get speedup info for using optimized parallel parameters
157  if (run_imp_settings.opt_parallel_params_setting.first) {
158  results_w_speedups.second.push_back(
159  GetAvgMedSpeedupOptPParams(
160  results_w_speedups.first,
161  std::string(run_eval::kSpeedupOptParParamsHeader) + " - " + std::string(run_eval::kAllRunsStr)));
162  }
163 
164  //get speedup when using templated number for loop iteration count
165  if (run_imp_settings.templated_iters_setting == run_environment::TemplatedItersSetting::kRunTemplatedAndNotTemplated) {
166  results_w_speedups.second.push_back(
167  GetAvgMedSpeedupLoopItersInTemplate(
168  results_w_speedups.first,
169  std::string(run_eval::kSpeedupLoopItersCountTemplate) + " - " + std::string(run_eval::kAllRunsStr)));
170  }
171 
172  //add speedups when using doubles and half precision compared to float to end of speedup data
173  //if speedup data exists
174  for (const auto& alt_speedup : alt_datatype_speedup) {
175  results_w_speedups.second.push_back(alt_speedup.second);
176  }
177 
178  //write output corresponding to results and speedups for all data types
179  WriteRunOutput(results_w_speedups, run_imp_settings, opt_imp_acc);
180 }

182 //write data for file corresponding to runs for a specified data type or across all data type
183 //includes results for each run as well as average and median speedup data across multiple runs
184 void EvaluateImpResults::WriteRunOutput(
185  const std::pair<MultRunData, std::vector<RunSpeedupAvgMedian>>& run_output,
186  const run_environment::RunImpSettings& run_imp_settings,
187  run_environment::AccSetting acceleration_setting) const
188 {
189  //get iterator to first run with success
190  //run_result corresponds to a std::optional object that contains run data
191  //if run successful and returns false if no data (indicating run not successful)
192  const auto first_success_run_iter =
193  std::find_if(run_output.first.cbegin(), run_output.first.cend(),
194  [](const auto& run_result) { return run_result.second; } );
195 
196  //check if there was at least one successful run
197  if (first_success_run_iter != run_output.first.cend()) {
198  //write results from default and optimized parallel parameters runs to csv file
199  //file name contains info about data type, parameter settings, and processor name if available
200  //only show data type string and acceleration string for runs using a single data type that are
201  //used for debugging (not multidata type results)
202  //get directory to store implementation results for specific implementation
203  const auto imp_results_fp = GetImpResultsPath();
204  //create any results directories if needed
205  if (!(std::filesystem::is_directory(imp_results_fp / run_eval::kImpResultsRunDataFolderName))) {
206  std::filesystem::create_directory(imp_results_fp / run_eval::kImpResultsRunDataFolderName);
207  }
208  if (!(std::filesystem::is_directory(imp_results_fp / run_eval::kImpResultsRunDataDefaultPParamsFolderName))) {
209  std::filesystem::create_directory(imp_results_fp / run_eval::kImpResultsRunDataDefaultPParamsFolderName);
210  }
211  if (!(std::filesystem::is_directory(imp_results_fp / run_eval::kImpResultsRunDataWSpeedupsFolderName))) {
212  std::filesystem::create_directory(imp_results_fp / run_eval::kImpResultsRunDataWSpeedupsFolderName);
213  }
214  if (!(std::filesystem::is_directory(imp_results_fp / run_eval::kImpResultsSpeedupsFolderName))) {
215  std::filesystem::create_directory(imp_results_fp / run_eval::kImpResultsSpeedupsFolderName);
216  }
217  //get file paths for run result and speedup files for implementation run
218  const std::filesystem::path opt_results_file_path{imp_results_fp / run_eval::kImpResultsRunDataFolderName /
219  std::filesystem::path(((run_imp_settings.run_name) ? std::string(run_imp_settings.run_name.value()) + "_" : "") +
220  std::string(run_eval::kRunResultsDescFileName) + std::string(run_eval::kCsvFileExtension))};
221  const std::filesystem::path default_params_results_file_path{imp_results_fp / run_eval::kImpResultsRunDataDefaultPParamsFolderName /
222  std::filesystem::path(((run_imp_settings.run_name) ? std::string(run_imp_settings.run_name.value()) + "_" : "") +
223  std::string(run_eval::kRunResultsDefaultPParamsDescFileName) + std::string(run_eval::kCsvFileExtension))};
224  const std::filesystem::path opt_results_w_speedup_file_path{imp_results_fp / run_eval::kImpResultsRunDataWSpeedupsFolderName /
225  std::filesystem::path(((run_imp_settings.run_name) ? std::string(run_imp_settings.run_name.value()) + "_" : "") +
226  std::string(run_eval::kRunResultsWSpeedupsDescFileName) + std::string(run_eval::kCsvFileExtension))};
227  const std::filesystem::path speedups_results_file_path{imp_results_fp / run_eval::kImpResultsSpeedupsFolderName /
228  std::filesystem::path(((run_imp_settings.run_name) ? std::string(run_imp_settings.run_name.value()) + "_" : "") +
229  std::string(run_eval::kSpeedupsDescFileName) + std::string(run_eval::kCsvFileExtension))};
230 
231  //initialize parallel params setting enum and vector of parallel params settings that are
232  //enabled in run
233  std::vector<run_environment::ParallelParamsSetting> parallel_param_settings{run_environment::ParallelParamsSetting::kDefault};
234  if (run_imp_settings.opt_parallel_params_setting.first) {
235  //add optimized parallel parameters settings if enabled
236  parallel_param_settings.push_back(run_environment::ParallelParamsSetting::kOptimized);
237  }
238 
239  //initialize map of ostringstreams for run data using default and optimized parallel parameters
240  std::map<run_environment::ParallelParamsSetting, std::ostringstream> run_data_sstr;
241  for (const auto& p_param_setting : parallel_param_settings) {
242  run_data_sstr[p_param_setting] = std::ostringstream();
243  }
244 
245  //initialize map of ostringstreams for speedup data with speedup headers on left and top
246  enum class SpeedupHeaderPlacement { kLeft, kTop };
247  std::map<SpeedupHeaderPlacement, std::ostringstream> speedups_w_headers_sstr;
248  speedups_w_headers_sstr[SpeedupHeaderPlacement::kLeft] = std::ostringstream();
249  speedups_w_headers_sstr[SpeedupHeaderPlacement::kTop] = std::ostringstream();
250 
251  //get headers from first successful run and write headers to top of output files
252  auto headers_in_order = first_success_run_iter->second->at(
253  run_environment::ParallelParamsSetting::kOptimized).HeadersInOrder();
254 
255  //get vector of speedup headers to use for evaluation across runs
256  //and also for run results
257  std::vector<std::string> speedup_headers;
258  for (const auto& speedup_header_data : run_output.second) {
259  speedup_headers.push_back(speedup_header_data.first);
260  }
261  //delete any speedup headers already in headers_in_order since not all
262  //of them may be included in first successful run
263  //then use vector of speedup headers to add to other headers for
264  //run resuls
265  for (const auto& speedup_header : speedup_headers) {
266  auto speedup_header_iter =
267  std::find(headers_in_order.begin(), headers_in_order.end(), speedup_header);
268  if (speedup_header_iter != headers_in_order.end()) {
269  headers_in_order.erase(speedup_header_iter);
270  }
271  }
272  //add speedup headers to the end of headers in order
273  headers_in_order.insert(
274  headers_in_order.end(), speedup_headers.begin(), speedup_headers.end());
275 
276  for (const auto& curr_header : headers_in_order) {
277  for (const auto& p_param_setting : parallel_param_settings) {
278  run_data_sstr[p_param_setting] << curr_header << ',';
279  }
280  }
281  for (const auto& p_param_setting : parallel_param_settings) {
282  run_data_sstr[p_param_setting] << std::endl;
283  }
284 
285  //write output for run on each input with each data type
286  //write data for default parallel parameters and for optimized parallel parameters
287  for (const auto& p_param_setting : parallel_param_settings) {
288  for (auto run_sig_data_iter=run_output.first.begin(); run_sig_data_iter != run_output.first.end(); run_sig_data_iter++) {
289  //if run not successful only have single set of output data from run
290  //don't write data if no data for run
291  if (run_sig_data_iter->second.has_value()) {
292  for (const auto& curr_header : headers_in_order) {
293  if (run_sig_data_iter->second->at(p_param_setting).IsData(curr_header))
294  {
295  run_data_sstr[p_param_setting] <<
296  run_sig_data_iter->second->at(p_param_setting).GetDataAsStr(curr_header) << ',';
297  }
298  else {
299  run_data_sstr[p_param_setting] << "No Data" << ',';
300  }
301  }
302  run_data_sstr[p_param_setting] << std::endl;
303  }
304  }
305  }
306 
307  //generate speedup results with headers on left side (saved to file with run results)
308  speedups_w_headers_sstr[SpeedupHeaderPlacement::kLeft] <<
309  "Speedup Results,Average Speedup,Median Speedup" << std::endl;
310  for (const auto& speedup : run_output.second) {
311  speedups_w_headers_sstr[SpeedupHeaderPlacement::kLeft] << speedup.first;
312  if ((speedup.second.contains(run_eval::MiddleValData::kAverage)) &&
313  (speedup.second.at(run_eval::MiddleValData::kAverage) > 0)) {
314  speedups_w_headers_sstr[SpeedupHeaderPlacement::kLeft] << ',' <<
315  speedup.second.at(run_eval::MiddleValData::kAverage) << ',' <<
316  speedup.second.at(run_eval::MiddleValData::kMedian);
317  }
318  else {
319  speedups_w_headers_sstr[SpeedupHeaderPlacement::kLeft] << ",,";
320  }
321  speedups_w_headers_sstr[SpeedupHeaderPlacement::kLeft] << std::endl;
322  }
323 
324  //generate speedup results with headers on top row (saved to separate "speedup" file)
325  speedups_w_headers_sstr[SpeedupHeaderPlacement::kTop] << ',';
326  for (const auto& speedup : run_output.second) {
327  speedups_w_headers_sstr[SpeedupHeaderPlacement::kTop] << speedup.first << ',';
328  }
329  for (const auto& speedup_desc_w_enum :
330  {std::pair<std::string_view, run_eval::MiddleValData>{"Average Speedup", run_eval::MiddleValData::kAverage},
331  std::pair<std::string_view, run_eval::MiddleValData>{"Median Speedup", run_eval::MiddleValData::kMedian}})
332  {
333  speedups_w_headers_sstr[SpeedupHeaderPlacement::kTop] << std::endl << speedup_desc_w_enum.first << ',';
334  for (const auto& speedup : run_output.second) {
335  if (speedup.second.contains(speedup_desc_w_enum.second)) {
336  speedups_w_headers_sstr[SpeedupHeaderPlacement::kTop] << speedup.second.at(speedup_desc_w_enum.second) << ',';
337  }
338  else {
339  speedups_w_headers_sstr[SpeedupHeaderPlacement::kTop] << ',';
340  }
341  }
342  }
343 
344  //write run results strings to output streams
345  //one results file contains speedup results, two contain run results
346  //(with and without optimized parallel parameters), and another
347  //contains run results followed by speedups
348  std::ofstream speedup_results_str{speedups_results_file_path};
349  speedup_results_str << speedups_w_headers_sstr[SpeedupHeaderPlacement::kTop].str();
350 
351  //write run results with and without optimized parallel parameters to files
352  std::map<run_environment::ParallelParamsSetting, std::ofstream> results_stream;
353  results_stream[run_environment::ParallelParamsSetting::kDefault] =
354  std::ofstream(default_params_results_file_path);
355  if (run_imp_settings.opt_parallel_params_setting.first) {
356  results_stream[run_environment::ParallelParamsSetting::kOptimized] =
357  std::ofstream(opt_results_file_path);
358  }
359  std::ofstream run_result_w_speedup_sstr(opt_results_w_speedup_file_path);
360  //write run results file with default parallel params
361  results_stream[run_environment::ParallelParamsSetting::kDefault] <<
362  run_data_sstr[run_environment::ParallelParamsSetting::kDefault].str();
363  if (run_imp_settings.opt_parallel_params_setting.first) {
364  //write run results file with optimized parallel params
365  results_stream[run_environment::ParallelParamsSetting::kOptimized] <<
366  run_data_sstr[run_environment::ParallelParamsSetting::kOptimized].str();
367  //add optimized run results to ostringstream for output containing run results and speedups
368  run_result_w_speedup_sstr << run_data_sstr[run_environment::ParallelParamsSetting::kOptimized].str() << std::endl;
369  }
370  else {
371  run_result_w_speedup_sstr << run_data_sstr[run_environment::ParallelParamsSetting::kDefault].str() << std::endl;
372  }
373 
374  //add speedups with headers on left to file containing run results and speedups
375  run_result_w_speedup_sstr << speedups_w_headers_sstr[SpeedupHeaderPlacement::kLeft].str();
376 
377  //close streams for writing to results files since file writing is done
378  speedup_results_str.close();
379  results_stream[run_environment::ParallelParamsSetting::kDefault].close();
380  results_stream[run_environment::ParallelParamsSetting::kOptimized].close();
381  run_result_w_speedup_sstr.close();
382 
383  //print location of output evaluation files to standard output
384  std::cout << "Input/settings/parameters info, detailed timings, and evaluation for each run and across runs in "
385  << opt_results_w_speedup_file_path << std::endl;
386  std::cout << "Run inputs and results in " << opt_results_file_path << std::endl;
387  std::cout << "Speedup results in " << speedups_results_file_path << std::endl;
388  std::cout << "Run inputs and results using default parallel parameters in "
389  << default_params_results_file_path << std::endl;
390 
391  //run evaluation across current and previous runs across architectures
392  //using run results and speedups saved from previous runs along with
393  //current run results
394  EvaluateAcrossRuns().operator()(
395  imp_results_fp, GetCombResultsTopText(), GetInputParamsShow(), speedup_headers);
396  }
397  else {
398  std::cout << "Error, no runs completed successfully" << std::endl;
399  }
400 }

402 //process results for runs with alternate acceleration from optimal acceleration and
403 //get speedup for each run and overall when using optimal acceleration compared to alternate accelerations
404 std::vector<RunSpeedupAvgMedian> EvaluateImpResults::GetAltAccelSpeedups(
405  MultRunDataWSpeedupByAcc& run_imp_results_by_acc_setting,
406  const run_environment::RunImpSettings& run_imp_settings,
407  size_t data_type_size,
408  run_environment::AccSetting fastest_acc) const
409 {
410  //set up mapping from acceleration type to description
411  const std::map<run_environment::AccSetting, std::string> acc_to_speedup_str{
412  {run_environment::AccSetting::kNone,
413  std::string(run_eval::kSpeedupCPUVectorization) + " - " + std::string(run_environment::kDataSizeToNameMap.at(data_type_size))},
414  {run_environment::AccSetting::kAVX256,
415  std::string(run_eval::kSpeedupVsAvx256Vectorization) + " - " + std::string(run_environment::kDataSizeToNameMap.at(data_type_size))}};
416 
417  if (run_imp_results_by_acc_setting.size() == 1) {
418  //no alternate run results
419  return {{acc_to_speedup_str.at(run_environment::AccSetting::kNone),
420  {}},
421  {acc_to_speedup_str.at(run_environment::AccSetting::kAVX256),
422  {}}};
423  }
424  else {
425  //initialize speedup/slowdown using alternate acceleration
426  std::vector<RunSpeedupAvgMedian> alt_acc_speedups;
427  for (auto& alt_acc_imp_results : run_imp_results_by_acc_setting) {
428  if ((alt_acc_imp_results.first != fastest_acc) && (acc_to_speedup_str.contains(alt_acc_imp_results.first))) {
429  //process results using alternate acceleration
430  //go through each result and replace initial run data with alternate implementation run data if alternate implementation run is faster
431  for (auto run_input_data = run_imp_results_by_acc_setting[fastest_acc].first.begin();
432  run_input_data != run_imp_results_by_acc_setting[fastest_acc].first.end();
433  run_input_data++)
434  {
435  if (run_input_data->second && alt_acc_imp_results.second.first[run_input_data->first])
436  {
437  const double init_result_time =
438  run_input_data->second->at(
439  run_environment::ParallelParamsSetting::kOptimized).GetDataAsDouble(run_eval::kOptimizedRuntimeHeader).value();
440  const double alt_acc_result_time =
441  alt_acc_imp_results.second.first[run_input_data->first]->at(
442  run_environment::ParallelParamsSetting::kOptimized).GetDataAsDouble(run_eval::kOptimizedRuntimeHeader).value();
443  if (alt_acc_result_time < init_result_time) {
444  run_input_data->second = alt_acc_imp_results.second.first[run_input_data->first];
445  }
446  }
447  }
448  //get speedup/slowdown using alternate acceleration compared to fastest implementation and store in speedup results
449  alt_acc_speedups.push_back(GetAvgMedSpeedupBaseVsTarget(
450  alt_acc_imp_results.second.first,

517 //get average and median speedup from vector of speedup values
518 std::map<run_eval::MiddleValData, double> EvaluateImpResults::GetAvgMedSpeedup(
519  const std::vector<double>& speedups_vect) const
520 {
521  const double average_speedup =
522  (std::accumulate(speedups_vect.cbegin(), speedups_vect.cend(), 0.0) / (double)speedups_vect.size());
523  auto speedups_vect_sorted = speedups_vect;
524  std::ranges::sort(speedups_vect_sorted);
525  const double median_speedup = ((speedups_vect_sorted.size() % 2) == 0) ?
526  (speedups_vect_sorted[(speedups_vect_sorted.size() / 2) - 1] + speedups_vect_sorted[(speedups_vect_sorted.size() / 2)]) / 2.0 :
527  speedups_vect_sorted[(speedups_vect_sorted.size() / 2)];
528  return {{run_eval::MiddleValData::kAverage, average_speedup},
529  {run_eval::MiddleValData::kMedian, median_speedup}};
530 }

585 //get average and median speedup of current runs compared to baseline data from file
586 std::vector<RunSpeedupAvgMedian> EvaluateImpResults::GetAvgMedSpeedupOverBaseline(
587  MultRunData& run_output,
588  std::string_view data_type_str,
589  const std::array<std::string_view, 2>& baseline_runtimes_path_desc) const
590 {
591  //define the start of the speedup info and runtime header in run data
592  //that is being compared for optimized and single thread implementations
593  const std::array<std::string_view, 2> baseline_opt_start_info_runtime_header{
594  "Speedup relative to", run_eval::kOptimizedRuntimeHeader};
595  const std::array<std::string_view, 2> baseline_s_thread_start_info_runtime_header{
596  "Single-Thread (Orig Imp) speedup relative to", run_eval::kSingleThreadRuntimeHeader};
597 
598  //vector for speedup data over baseline for optimized and single thread
599  //implementations
600  std::vector<RunSpeedupAvgMedian> speedup_data;
601 
602  //get speedup of current run compared to baseline optimized and
603  //single thread implementations
604  for (const auto& start_info_runtime_header :
605  {baseline_opt_start_info_runtime_header,
606  baseline_s_thread_start_info_runtime_header})
607  {
608  //get baseline run data for runs according to current runtime header
609  const auto baseline_run_data =
610  GetBaselineRuntimeData(baseline_runtimes_path_desc, start_info_runtime_header.at(1));
611  if (baseline_run_data) {

614  " " + baseline_run_data->first + " - " + std::string(data_type_str);
615  for (auto run_input_data_iter=run_output.begin();
616  run_input_data_iter != run_output.end();
617  run_input_data_iter++)
618  {
619  //compute speedup for each run compared to baseline runtime and add
620  //to vector of speedups as well as to run data
621  if (run_input_data_iter->second) {
622  speedups_vect.push_back(
623  std::stod(baseline_run_data->second.at(run_input_data_iter->first)) /
624  (run_input_data_iter->second->at(
625  run_environment::ParallelParamsSetting::kOptimized).GetDataAsDouble(
626  start_info_runtime_header.at(1)).value()));
627  for (auto& run_data : run_output.at(run_input_data_iter->first).value()) {
628  run_data.second.AddDataWHeader(speedup_header, speedups_vect.back());
629  }
630  }
631  }
632  if (!(speedups_vect.empty())) {
633  //get average and median speedups across all runs compared to baseline runtimes
634  speedup_data.push_back({speedup_header, GetAvgMedSpeedup(speedups_vect)});
635  }
636  }
637  }
638 
639  return speedup_data;

682 //get average and median speedup between base and target runtime data and also add
683 //speedup for each target runtime data run as compared to corresponding base run
684 RunSpeedupAvgMedian EvaluateImpResults::GetAvgMedSpeedupBaseVsTarget(
685  MultRunData& run_output_base,
686  MultRunData& run_output_target,
687  std::string_view speedup_header,
688  BaseTargetDiff base_target_diff) const
689 {
690  std::vector<double> speedups_vect;
691  //go through all base runtime data
692  for (auto run_input_data_iter_base=run_output_base.begin();
693  run_input_data_iter_base != run_output_base.end();
694  run_input_data_iter_base++)
695  {
696  InputSignature base_in_sig_adjusted(run_input_data_iter_base->first);
697  //go through all target runtime data and find speedup for data
698  //that corresponds with current base runtime data
699  for (auto run_input_data_iter_target=run_output_target.begin();
700  run_input_data_iter_target != run_output_target.end();
701  run_input_data_iter_target++)
702  {
703  InputSignature target_in_sig_adjusted(run_input_data_iter_target->first);
704  if (base_target_diff == BaseTargetDiff::kDiffDatatype) {
705  //remove datatype from input signature if different datatype
706  //between base and target output
707  base_in_sig_adjusted.RemoveDatatypeSetting();
708  target_in_sig_adjusted.RemoveDatatypeSetting();
709  }
710  else if (base_target_diff == BaseTargetDiff::kDiffTemplatedSetting) {
711  //remove templated setting from input signature if different template
712  //setting between base and target output
713  base_in_sig_adjusted.RemoveTemplatedLoopIterSetting();
714  target_in_sig_adjusted.RemoveTemplatedLoopIterSetting();
715  }
716  if (base_in_sig_adjusted == target_in_sig_adjusted) {
717  //compute speedup between corresponding base and target data and
718  //add it to vector of speedups across all data
719  if (run_input_data_iter_base->second.has_value() &&
720  run_input_data_iter_target->second.has_value())
721  {
722  speedups_vect.push_back(
723  run_input_data_iter_base->second->at(
724  run_environment::ParallelParamsSetting::kOptimized).GetDataAsDouble(
725  run_eval::kOptimizedRuntimeHeader).value() /
726  run_input_data_iter_target->second->at(
727  run_environment::ParallelParamsSetting::kOptimized).GetDataAsDouble(
728  run_eval::kOptimizedRuntimeHeader).value());
729 
730  //add speedup data to corresponding base and target data
731  run_input_data_iter_base->second->at(
732  run_environment::ParallelParamsSetting::kOptimized).AddDataWHeader(
733  std::string(speedup_header), speedups_vect.back());
734  run_input_data_iter_target->second->at(
735  run_environment::ParallelParamsSetting::kOptimized).AddDataWHeader(
736  std::string(speedup_header), speedups_vect.back());
737  }
738  }
739  }
740  }
741  if (!(speedups_vect.empty())) {
742  //get average and median speedup from speedup vector with all speedup
743  //data and return as pair with header describing speedup
744  return {std::string(speedup_header), GetAvgMedSpeedup(speedups_vect)};
745  }
746  return {std::string(speedup_header), {}};
747 }
