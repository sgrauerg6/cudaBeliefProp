94 //run and evaluate optimized belief propagation implementation on evaluation stereo set
95 //specified by NUM_INPUT
96 //data type used in implementation specified by T
97 //bp implemenation optimization specified by OPT_IMP_ACCEL
98 //evaluation stereo set to run implementation on specified by NUM_INPUT
99 template<RunData_t T, run_environment::AccSetting OPT_IMP_ACCEL, unsigned int NUM_INPUT>
100 MultRunData RunEvalBPImpOnInput<T, OPT_IMP_ACCEL, NUM_INPUT>::operator()(
101  const run_environment::RunImpSettings& run_imp_settings)
102 {
103  //set up BP settings for current run
104  alg_settings_.num_disp_vals = beliefprop::kStereoSetsToProcess[NUM_INPUT].num_disp_vals;
105  alg_settings_.disc_k_bp = (float)alg_settings_.num_disp_vals / 7.5f;
106 
107  //initialize run results across multiple implementations
108  MultRunData run_results;
109 
110  //set up unoptimized single threaded bp stereo implementation
111  run_bp_stereo_single_thread_ =
112  std::make_unique<RunBpStereoCPUSingleThread<T, beliefprop::kStereoSetsToProcess[NUM_INPUT].num_disp_vals, run_environment::AccSetting::kNone>>();
113 
114  //set up and run bp stereo using optimized implementation (optimized CPU and CUDA implementations supported)
115  //as well as unoptimized implementation for comparison
116  //run optimized implementation with and/or without disparity value count templated depending on setting
117  //for running implementation with disparity values templated
118  if (run_imp_settings.templated_iters_setting != run_environment::TemplatedItersSetting::kRunOnlyNonTemplated) {
119  run_opt_bp_num_iters_templated_ =
120  std::make_unique<RunBpOptimized<T, beliefprop::kStereoSetsToProcess[NUM_INPUT].num_disp_vals, OPT_IMP_ACCEL>>();
121  constexpr bool run_w_loop_iters_templated{true};
122  InputSignature input_sig(sizeof(T), NUM_INPUT, run_w_loop_iters_templated);
123  run_results.insert({input_sig, (this->RunEvalBenchmark(run_imp_settings, run_w_loop_iters_templated))});
124  }
125  if (run_imp_settings.templated_iters_setting != run_environment::TemplatedItersSetting::kRunOnlyTempated) {
126  run_opt_bp_num_iters_no_template_ =
127  std::make_unique<RunBpOptimized<T, 0, OPT_IMP_ACCEL>>();
128  constexpr bool run_w_loop_iters_templated{false};
129  InputSignature input_sig(sizeof(T), NUM_INPUT, run_w_loop_iters_templated);
130  run_results.insert({input_sig, this->RunEvalBenchmark(run_imp_settings, run_w_loop_iters_templated)});
131  }
132 
133  //return bp run results across multiple implementations
134  return run_results;
135 }

161 //run and compare output disparity maps using the given optimized and single-threaded stereo implementations
162 //on the reference and test images specified by numStereoSet
163 //run only optimized implementation if run_opt_imp_only is true
164 template<RunData_t T, run_environment::AccSetting OPT_IMP_ACCEL, unsigned int NUM_INPUT>
165 std::optional<RunData> RunEvalBPImpOnInput<T, OPT_IMP_ACCEL, NUM_INPUT>::RunImpsAndCompare(
166  std::shared_ptr<ParallelParams> parallel_params, bool run_opt_imp_only, bool run_imp_templated_loop_iters) const
167 {
168  //get properties of input stereo set from stereo set number
169  BpFileHandling bp_file_settings(std::string(beliefprop::kStereoSetsToProcess[NUM_INPUT].name));
170  const std::array<filepathtype, 2> ref_test_image_path{bp_file_settings.RefImagePath(), bp_file_settings.TestImagePath()};
171 
172  //get number of implementations to run output disparity map file path(s)
173  //if run_opt_imp_only is false, run single-threaded implementation in addition to optimized implementation
174  const unsigned int num_imps_run{run_opt_imp_only ? 1u : 2u};
175  std::array<filepathtype, 2> output_disp;
176  for (unsigned int i=0; i < num_imps_run; i++) {
177  output_disp[i] = bp_file_settings.GetCurrentOutputDisparityFilePathAndIncrement();
178  }
179 
180  //get optimized implementation description and write info about run to std::cout stream
181  const std::string opt_imp_run_description{run_imp_templated_loop_iters ?
182  run_opt_bp_num_iters_templated_->BpRunDescription() :
183  run_opt_bp_num_iters_no_template_->BpRunDescription()};
184  std::cout << "Running belief propagation on reference image " << ref_test_image_path[0] << " and test image "
185  << ref_test_image_path[1] << " on " << opt_imp_run_description;
186  if (!run_opt_imp_only) {
187  std::cout << " and " << run_bp_stereo_single_thread_->BpRunDescription();
188  }
189  std::cout << std::endl;
190 
191  //run optimized implementation and retrieve structure with runtime and output disparity map
192  std::map<run_environment::AccSetting, std::optional<ProcessStereoSetOutput>> run_output;
193  if (run_imp_templated_loop_iters) {
194  run_output[OPT_IMP_ACCEL] = run_opt_bp_num_iters_templated_->operator()(
195  {ref_test_image_path[0].string(), ref_test_image_path[1].string()}, alg_settings_, *parallel_params);
196  }
197  else {
198  run_output[OPT_IMP_ACCEL] = run_opt_bp_num_iters_no_template_->operator()(
199  {ref_test_image_path[0].string(), ref_test_image_path[1].string()}, alg_settings_, *parallel_params);
200  }
201 
202  //check if error in run
203  RunData run_data;
204  run_data.AddDataWHeader(std::string(beliefprop::kAccelerationDescHeader), opt_imp_run_description);
205  if (!(run_output[OPT_IMP_ACCEL])) {
206  return {};
207  }
208 
209  //append implementation run data to evaluation run data
210  run_data.AppendData(run_output[OPT_IMP_ACCEL]->run_data);
211  run_data.AddDataWHeader(
212  std::string(run_eval::kOptimizedRuntimeHeader), run_output[OPT_IMP_ACCEL]->run_time.count());
213 
214  //save resulting disparity map
215  run_output[OPT_IMP_ACCEL]->out_disparity_map.SaveDisparityMap(
216  output_disp[0].string(), beliefprop::kStereoSetsToProcess[NUM_INPUT].scale_factor);
217 
218  //check if only running optimized implementation or if also running single-threaded implementation
219  if (!run_opt_imp_only) {
220  //check if running single threaded implementation or using stored results from previous run
221  if ((!(bpSingleThread::kRunSingleThreadOnceForSet)) || (!(bpSingleThread::single_thread_run_output.contains(ref_test_image_path))))
222  {
223  //run single-threaded implementation and retrieve structure with runtime and output disparity map
224  run_output[run_environment::AccSetting::kNone] = run_bp_stereo_single_thread_->operator()(
225  {ref_test_image_path[0].string(), ref_test_image_path[1].string()}, alg_settings_, *parallel_params);
226  if (!(run_output[run_environment::AccSetting::kNone])) {
227  return {};
228  }
229 
230  //save resulting disparity map
231  run_output[run_environment::AccSetting::kNone]->out_disparity_map.SaveDisparityMap(
232  output_disp[1].string(), beliefprop::kStereoSetsToProcess[NUM_INPUT].scale_factor);
233 
234  //save run data and result if setting to only run single thread implementation once for each set
235  if (bpSingleThread::kRunSingleThreadOnceForSet) {
236  bpSingleThread::single_thread_run_output[ref_test_image_path] =
237  {run_output[run_environment::AccSetting::kNone]->run_time, run_output[run_environment::AccSetting::kNone]->run_data, output_disp[1]};
238  }
239  }
240  else {
241  run_output[run_environment::AccSetting::kNone] = ProcessStereoSetOutput();
242  //retrieve stored results from previous run on single threaded implementation
243  run_output[run_environment::AccSetting::kNone]->run_time = std::get<0>(bpSingleThread::single_thread_run_output[ref_test_image_path]);
244  run_output[run_environment::AccSetting::kNone]->run_data = std::get<1>(bpSingleThread::single_thread_run_output[ref_test_image_path]);
245  run_output[run_environment::AccSetting::kNone]->out_disparity_map = DisparityMap<float>(
246  std::get<2>(bpSingleThread::single_thread_run_output[ref_test_image_path]).string(),
247  beliefprop::kStereoSetsToProcess[NUM_INPUT].scale_factor);
248  }
249  //append run data for single thread run to evaluation run data
250  run_data.AppendData(run_output[run_environment::AccSetting::kNone]->run_data);
251  }
252 
253  //write location of output disparity maps for each implementation to std::cout
254  for (unsigned int i = 0; i < num_imps_run; i++) {
255  const std::string run_description{(i == 0) ? opt_imp_run_description : run_bp_stereo_single_thread_->BpRunDescription()};
256  std::cout << "Output disparity map from " << run_description << " run at " << output_disp[i] << std::endl;
257  }
258  std::cout << std::endl;
259 
260  //compare resulting disparity maps with ground truth and to each other
261  const filepathtype ground_truth_disp{bp_file_settings.GroundTruthDisparityFilePath()};
262  DisparityMap<float> ground_truth_disparity_map(
263  ground_truth_disp.string(), beliefprop::kStereoSetsToProcess[NUM_INPUT].scale_factor);
264  run_data.AddDataWHeader(opt_imp_run_description + " output vs. Ground Truth result", std::string());
265  run_data.AppendData(run_output[OPT_IMP_ACCEL]->out_disparity_map.OutputComparison(
266  ground_truth_disparity_map, BpEvaluationParameters()).AsRunData());
267  if (!run_opt_imp_only) {
268  run_data.AddDataWHeader(run_bp_stereo_single_thread_->BpRunDescription() + " output vs. Ground Truth result", std::string());
269  run_data.AppendData(run_output[run_environment::AccSetting::kNone]->out_disparity_map.OutputComparison(
270  ground_truth_disparity_map, BpEvaluationParameters()).AsRunData());
271  run_data.AddDataWHeader(
272  opt_imp_run_description + " output vs. " + run_bp_stereo_single_thread_->BpRunDescription() + " result", std::string());
273  run_data.AppendData(run_output[OPT_IMP_ACCEL]->out_disparity_map.OutputComparison(
274  run_output[run_environment::AccSetting::kNone]->out_disparity_map, BpEvaluationParameters()).AsRunData());
275  }
276 
277  //return structure indicating that run succeeded along with data from run
278  return run_data;
279 }
