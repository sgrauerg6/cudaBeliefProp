67  float lambda_bp, float data_k_bp, unsigned int bp_settings_disp_vals)
68 {
69  //get the x and y indices for the current CUDA thread

71  const unsigned int y_val = blockIdx.y * blockDim.y + threadIdx.y;
72

76  if (util_functs::WithinImageBounds(

78  {
79  beliefprop::InitializeBottomLevelDataPixel<T, DISP_VALS>(x_val, y_val,
80  current_bp_level, image_1_pixels_device,
81  image_2_pixels_device, data_cost_stereo_checkerboard_0,
