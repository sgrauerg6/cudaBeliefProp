38 //initialize the "data cost" for each possible disparity between the two full-sized input images ("bottom" of the image pyramid)
39 //the image data is stored in the CUDA arrays image1PixelsTextureBPStereo and image2PixelsTextureBPStereo
40 template<RunData_t T, unsigned int DISP_VALS>
41 __global__ void InitializeBottomLevelData(
42  beliefprop::BpLevelProperties current_bp_level,
43  float* image_1_pixels_device, float* image_2_pixels_device,
44  T* data_cost_stereo_checkerboard_0, T* data_cost_stereo_checkerboard_1,
45  float lambda_bp, float data_k_bp, unsigned int bp_settings_disp_vals)
46 {
47  //get the x and y indices for the current CUDA thread
48  const unsigned int x_val = blockIdx.x * blockDim.x + threadIdx.x;
49  const unsigned int y_val = blockIdx.y * blockDim.y + threadIdx.y;
50 
51  //get the x value within the current "checkerboard"
52  const unsigned int x_checkerboard = x_val / 2;
53 
54  if (util_functs::WithinImageBounds(
55  x_checkerboard, y_val, current_bp_level.width_level_, current_bp_level.height_level_))
56  {
57  beliefprop::InitializeBottomLevelDataPixel<T, DISP_VALS>(x_val, y_val,
58  current_bp_level, image_1_pixels_device,
59  image_2_pixels_device, data_cost_stereo_checkerboard_0,
60  data_cost_stereo_checkerboard_1, lambda_bp,
61  data_k_bp, bp_settings_disp_vals);
62  }
63 }

116 //kernel function to run the current iteration of belief propagation in parallel using the
117 //checkerboard update method where half the pixels in the "checkerboard"
118 //scheme retrieve messages from each 4-connected neighbor and then update
119 //their message based on the retrieved messages and the data cost
120 template<RunData_t T, unsigned int DISP_VALS>
121 __global__ void RunBPIterationUsingCheckerboardUpdates(
122  beliefprop::CheckerboardPart checkerboard_to_update, beliefprop::BpLevelProperties current_bp_level,
123  T* data_cost_checkerboard_0, T* data_cost_checkerboard_1,
124  T* message_u_checkerboard_0, T* message_d_checkerboard_0,
125  T* message_l_checkerboard_0, T* message_r_checkerboard_0,
126  T* message_u_checkerboard_1, T* message_d_checkerboard_1,
127  T* message_l_checkerboard_1, T* message_r_checkerboard_1,
128  float disc_k_bp, bool data_aligned, unsigned int bp_settings_disp_vals)
129 {
130  //get the x and y indices for the current CUDA thread
131  const unsigned int x_val = blockIdx.x * blockDim.x + threadIdx.x;
132  const unsigned int y_val = blockIdx.y * blockDim.y + threadIdx.y;
133 
134  if (util_functs::WithinImageBounds(
135  x_val, y_val, current_bp_level.width_level_/2, current_bp_level.height_level_))
136  {
137  beliefprop::RunBPIterationUsingCheckerboardUpdatesKernel<T, T, DISP_VALS>(
138  x_val, y_val, checkerboard_to_update, current_bp_level,
139  data_cost_checkerboard_0, data_cost_checkerboard_1,
140  message_u_checkerboard_0, message_d_checkerboard_0,
141  message_l_checkerboard_0, message_r_checkerboard_0,
142  message_u_checkerboard_1, message_d_checkerboard_1,
143  message_l_checkerboard_1, message_r_checkerboard_1,
144  disc_k_bp, 0, data_aligned, bp_settings_disp_vals);
145  }
146 }
