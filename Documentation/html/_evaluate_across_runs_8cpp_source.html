37 //process runtime and speedup data across multiple runs (likely on different
38 //architectures) from csv files corresponding to each run and write results to
39 //file where the runtimes and speedups for each run are shown in a single file
40 //and where the runs are displayed left to right from fastest to slowest
41 void EvaluateAcrossRuns::operator()(
42  const std::filesystem::path& imp_results_file_path,
43  const std::vector<std::string>& eval_across_runs_top_text,
44  const std::vector<std::string>& eval_across_runs_in_params_show) const
45 {
46  //retrieve names of runs with results
47  //run names usually correspond to architecture of run
48  const std::vector<std::string> run_names =
49  GetRunNames(imp_results_file_path);
50 
51  //initialize vector of run results with speedups for each run name
52  std::map<std::string, RunResultsSpeedups> run_results_by_name;
53  for (const auto& run_name : run_names) {
54  run_results_by_name.insert(
55  {run_name, RunResultsSpeedups(imp_results_file_path, run_name)});
56  }
57 
58  //genereation data mappings for evaluation including run results and speedups
59  //for each run, mapping from input signature to runtime, and run inputs to
60  //parameters to display in evaluation across runs as well as speedup headers
61  //in order
62  const EvalAcrossRunsData eval_data =
63  GenEvalAcrossRunsData(
64  run_results_by_name,
65  eval_across_runs_in_params_show);
66 
67  //get run names in order from fastest to slowest based on first speedup
68  //header
69  std::vector<std::string> run_names_ordered =
70  OrderedRunNames(eval_data);
71 
72  //add any run names not in ordered run names (due to not having data
73  //corresponding to first speedup header) to end of ordered runs
74  std::copy_if(
75  run_names.begin(),
76  run_names.end(),
77  std::back_inserter(run_names_ordered),
78  [&run_names_ordered](const auto& run_name) {
79  return (
80  std::find(
81  run_names_ordered.begin(),
82  run_names_ordered.end(),
83  run_name) == run_names_ordered.end());
84  });
85 
86  //write output evaluation across runs to file
87  WriteEvalAcrossRunsToFile(
88  imp_results_file_path,
89  eval_across_runs_top_text,
90  eval_across_runs_in_params_show,
91  eval_data,
92  run_names_ordered);
93 }

103 EvalAcrossRunsData EvaluateAcrossRuns::GenEvalAcrossRunsData(
104  const std::map<std::string, RunResultsSpeedups>& run_results_by_name,
105  const std::vector<std::string>& eval_across_runs_in_params_show) const
106 {
107  EvalAcrossRunsData eval_data;
108  for (const auto& [run_name, run_results] : run_results_by_name)
109  {
110  eval_data.speedup_results_name_to_data.insert(
111  {run_name,
112  run_results.Speedups()});
113  eval_data.input_to_runtime_across_archs.insert(
114  {run_name,
115  run_results.InputsToKeyVal(run_eval::kOptimizedRuntimeHeader)});
116  const auto& inputs_to_runtimes = run_results.InputsToKeyVal(
117  run_eval::kOptimizedRuntimeHeader);
118  //go through inputs for current run results
119  //need to go through every run so that all inputs in every run are included
120  for (const auto& [input_sig, _] : inputs_to_runtimes) {
121  //check if input already addded to set of inputs
122  if (!(eval_data.inputs_to_params_disp_ordered.contains(input_sig))) {
123  eval_data.inputs_to_params_disp_ordered.insert(
124  {input_sig, std::vector<std::string>()});
125  //add input parameters to display in evaluation across runs
126  for (const auto& disp_param : eval_across_runs_in_params_show) {
127  eval_data.inputs_to_params_disp_ordered.at(input_sig).push_back(
128  run_results.DataForInput(input_sig).at(disp_param));
129  }
130  }
131  }
132  //go through speedups for run and add to speedup headers if not already
133  //included
134  const auto run_speedups_ordered = run_results.SpeedupHeadersOrder();
135  for (auto run_speedup_iter = run_speedups_ordered.cbegin();
136  run_speedup_iter < run_speedups_ordered.cend();
137  run_speedup_iter++)
138  {
139  //ignore speedup if all whitespace
140  if (std::all_of(
141  run_speedup_iter->begin(),
142  run_speedup_iter->end(),
143  [](unsigned char c){ return std::isspace(c); }))
144  {
145  continue;
146  }
147  //check if speedup in run is included in current evaluation speedups and
148  //add it in expected position in evaluation speedups if not
149  if (std::none_of(eval_data.speedup_headers.cbegin(),
150  eval_data.speedup_headers.cend(),
151  [run_speedup_iter](const auto& header){
152  return (header == *run_speedup_iter);
153  }))
154  {
155  //get relative position of previous ordered speedup header
156  auto iter_prev_ordered_header = run_speedup_iter;
157  while (iter_prev_ordered_header != run_speedups_ordered.cbegin())
158  {
159  iter_prev_ordered_header--;
160  if (std::find(eval_data.speedup_headers.cbegin(),
161  eval_data.speedup_headers.cend(),
162  *iter_prev_ordered_header) !=
163  eval_data.speedup_headers.cend())
164  {
165  break;
166  }
167  }
168 
169  //add speedup header in front of previous ordered speedup header if
170  //not first speedup header for run and previous ordered header exists
171  //in speedup headers
172  if (iter_prev_ordered_header != run_speedups_ordered.cbegin())
173  {
174  //find position in evaluation speedups of previous ordered header
175  //and add new header in front of it
176  eval_data.speedup_headers.insert(
177  (std::find(eval_data.speedup_headers.cbegin(),
178  eval_data.speedup_headers.cend(),
179  *iter_prev_ordered_header) + 1),
180  *run_speedup_iter);
181  }
182  else {
183  //add speedup header to front of evaluation speedup headers if no
184  //previous ordered header in speedups for run
185  eval_data.speedup_headers.insert(
186  eval_data.speedup_headers.cbegin(),
187  *run_speedup_iter);
188  }
189  }
190  }
191  }
192 
193  //return resulting evaluation data
194  return eval_data;
195 }

307 void EvaluateAcrossRuns::WriteEvalAcrossRunsToFile(
308  const std::filesystem::path& imp_results_file_path,
309  const std::vector<std::string>& eval_across_runs_top_text,
310  const std::vector<std::string>& eval_across_runs_in_params_show,
311  const EvalAcrossRunsData& eval_data,
312  const std::vector<std::string>& run_names_ordered) const
313 {
314  //write results across architectures to output file
315  //file path for evaluation across runs
316  const std::filesystem::path results_across_run_fp =
317  imp_results_file_path /
318  (std::string(run_eval::kEvalAcrossRunsFileName) +
319  std::string(run_eval::kCsvFileExtension));
320 
321  //initialize output stream for file showing evaluation across runs
322  std::ofstream eval_results_across_run_str(results_across_run_fp);
323 
324  //add text to display on top of results across architecture comparison file
325  for (const auto& comp_file_top_text_line : eval_across_runs_top_text) {
326  eval_results_across_run_str << comp_file_top_text_line << std::endl;
327  }
328  eval_results_across_run_str << std::endl;
329 
330  //write out the name of each input parameter to be displayed
331  for (const auto& input_param_disp_header : eval_across_runs_in_params_show) {
332  eval_results_across_run_str << input_param_disp_header << ',';
333  }
334 
335  //write all the run names in order from fastest to slowest
336  for (const auto& run_name : run_names_ordered) {
337  eval_results_across_run_str << run_name << ',';
338  }
339  eval_results_across_run_str << std::endl;
340 
341  //write evaluation stereo set info, bp parameters, and total runtime for
342  //optimized bp implementation across each run in the evaluation
343  for (const auto& [input_sig, params_display_ordered] :
344  eval_data.inputs_to_params_disp_ordered)
345  {
346  for (const auto& param_val_disp : params_display_ordered) {
347  eval_results_across_run_str << param_val_disp << ',';
348  }
349  for (const auto& run_name : run_names_ordered)
350  {
351  if (eval_data.input_to_runtime_across_archs.at(
352  run_name).contains(input_sig))
353  {
354  eval_results_across_run_str <<
355  eval_data.input_to_runtime_across_archs.at(run_name).at(input_sig);
356  }
357  eval_results_across_run_str << ',';
358  }
359  eval_results_across_run_str << std::endl;
360  }
361  eval_results_across_run_str << std::endl;
362 
363  //write average speedup results for each run that correspond to a number of
364  //different evaluations of runtimes compared to a baseline
365  eval_results_across_run_str << "Average Speedups" << std::endl;
366  for (const auto& speedup_header : eval_data.speedup_headers) {
367  //don't process if header is empty
368  if (!(speedup_header.empty())) {
369  eval_results_across_run_str << speedup_header << ',';
370  //add empty cell for each input parameter after the first that's
371  //displayed so speedup shown in the same column same line as runtime
372  for (size_t i = 1; i < eval_across_runs_in_params_show.size(); i++) {
373  eval_results_across_run_str << ',';
374  }
375  //write speedup for each run in separate cells in horizontal direction
376  //where each column corresponds to a different evaluation run
377  for (const auto& run_name : run_names_ordered) {
378  if (eval_data.speedup_results_name_to_data.at(run_name).contains(
379  speedup_header))
380  {
381  eval_results_across_run_str <<
382  eval_data.speedup_results_name_to_data.at(run_name).at(
383  speedup_header).at(0) << ',';
384  }
385  else {
386  eval_results_across_run_str << ',';
387  }
388  }
389  //continue to next row of table to write data for next speedup result
390  eval_results_across_run_str << std::endl;
391  }
392  }
393 
394  //go through each speedup and display runs ordered from highest speedup to lowest
395  //speedup
396  eval_results_across_run_str << std::endl;
397  eval_results_across_run_str << "Runs ordered by speedup (highest to lowest)" << std::endl;
398  for (const auto& speedup_header : eval_data.speedup_headers) {
399  //don't process if header is empty
400  if (!(speedup_header.empty())) {
401  eval_results_across_run_str << speedup_header << ',';
402  //get runs ordered by speedup
403  const auto ordered_runs_speedup =
404  OrderedRunNames(
405  eval_data,
406  speedup_header);
407  for (const auto& run_name : ordered_runs_speedup) {
408  const auto speedup =
409  eval_data.speedup_results_name_to_data.at(run_name).at(
410  speedup_header).at(0);
411  eval_results_across_run_str << run_name << " - " << std::setprecision(3)
412  << std::fixed << std::stof(speedup) << ',';
413  }
414  }
415  //continue to next row of table to write data for next speedup result
416  eval_results_across_run_str << std::endl;
417  }
418 
419  //write location of evaluation results across runs to output console
420  std::cout << "Evaluation of results across all runs in "
421  << results_across_run_fp << std::endl;
422 }

426 std::vector<std::string> EvaluateAcrossRuns::GetRunNames(
427  const std::filesystem::path& imp_results_file_path) const
428 {
429  //iterate through all run results files
430  //create directory iterator with path to run results files
431  std::filesystem::directory_iterator results_files_iter =
432  std::filesystem::directory_iterator(
433  imp_results_file_path / run_eval::kImpResultsRunDataFolderName);
434  std::vector<std::string> run_names;
435  for (const auto& results_fp : results_files_iter)
436  {
437  //get run name from run results file name
438  //and add to vector of run names
439  std::string file_name_no_ext = results_fp.path().stem();
440  if (file_name_no_ext.ends_with(
441  "_" + std::string(run_eval::kRunResultsDescFileName)))
442  {
443  const std::string run_name =
444  file_name_no_ext.substr(
445  0,
446  file_name_no_ext.find(
447  "_" + std::string(run_eval::kRunResultsDescFileName)));
448  run_names.push_back(run_name);
449  }
450  }
451 
452  //remove run name from runs to evaluate across runs if no speedup data
453  //example where this could be the case is baseline data that is used for
454  //comparison with other runs
455  for (auto run_names_iter = run_names.cbegin();
456  run_names_iter != run_names.cend();)
457  {
458  const std::filesystem::path run_speedup_fp =
459  imp_results_file_path / run_eval::kImpResultsSpeedupsFolderName /
460  (std::string(*run_names_iter) + '_' +
461  std::string(run_eval::kSpeedupsDescFileName) +
462  std::string(run_eval::kCsvFileExtension));
463  //remove run from evaluation if no valid speedup data file that corresponds
464  //to run results data file
465  if ((!(std::filesystem::exists(run_speedup_fp))) ||
466  (!(std::filesystem::is_regular_file(run_speedup_fp))))
467  {
468  run_names_iter =
469  run_names.erase(std::ranges::find(run_names, *run_names_iter));
470  }
471  else {
472  run_names_iter++;
473  }
474  }
475 
476  return run_names;
477 }
