37 //process runtime and speedup data across multiple runs (likely on different architectures)
38 //from csv files corresponding to each run and
39 //write results to file where the runtimes and speedups for each run are shown in a single file
40 //where the runs are displayed from fastest to slowest
41 void EvaluateAcrossRuns::operator()(
42  const std::filesystem::path& imp_results_file_path,
43  const std::vector<std::string>& eval_across_runs_top_text,
44  const std::vector<std::string>& eval_across_runs_in_params_show,
45  const std::vector<std::string>& speedup_headers) const
46 {
47  //initialize speedup headers for output results across runs
48  auto speedup_headers_eval = speedup_headers;
49 
50  //retrieve names of runs with results
51  //run names usually correspond to architecture of run
52  const std::vector<std::string> run_names =
53  GetRunNames(imp_results_file_path);
54 
55  //initialize vector of run results with speedups for each run name
56  std::map<std::string, RunResultsSpeedups> run_results_by_name;
57  for (const auto& run_name : run_names) {
58  run_results_by_name.insert(
59  {run_name, RunResultsSpeedups(imp_results_file_path, run_name)});
60  }
61 
62  //get header to data of run results and speedups for each run, mapping from
63  //input signature to runtime, and run inputs to parameters to display in
64  //evaluation across runs
65  std::map<std::string, std::map<std::string, std::vector<std::string>>> speedup_results_name_to_data;
66  std::map<std::string, std::map<InputSignature, std::string>> input_to_runtime_across_archs;
67  std::map<InputSignature, std::vector<std::string>> inputs_to_params_disp_ordered;
68  for (const auto& [run_name, run_results] : run_results_by_name) {
69  speedup_results_name_to_data.insert(
70  {run_name,
71  run_results.Speedups()});
72  input_to_runtime_across_archs.insert(
73  {run_name,
74  run_results.InputsToKeyVal(run_eval::kOptimizedRuntimeHeader)});
75  const auto& inputs_to_runtimes = run_results.InputsToKeyVal(
76  run_eval::kOptimizedRuntimeHeader);
77  //go through inputs for current run results
78  //need to go through every run so that all inputs in every run are included
79  for (const auto& [input_sig, runtime] : inputs_to_runtimes) {
80  //check if input already addded to set of inputs
81  if (!(inputs_to_params_disp_ordered.contains(input_sig))) {
82  inputs_to_params_disp_ordered.insert({input_sig, std::vector<std::string>()});
83  //add input parameters to display in evaluation across runs
84  for (const auto& disp_param : eval_across_runs_in_params_show) {
85  inputs_to_params_disp_ordered.at(input_sig).push_back(
86  run_results.DataForInput(input_sig).at(disp_param));
87  }
88  }
89  }
90  //go through speedups for run and add to speedup headers if not already included
91  const auto run_speedups_ordered = run_results.SpeedupHeadersOrder();
92  for (auto run_speedup_iter = run_speedups_ordered.cbegin();
93  run_speedup_iter < run_speedups_ordered.cend();
94  run_speedup_iter++)
95  {
96  //check if speedup in run is included in current evaluation speedups and add it
97  //in expected position in evaluation speedups if not
98  if (std::all_of(speedup_headers_eval.cbegin(),
99  speedup_headers_eval.cend(),
100  [run_speedup_iter](const auto& header){
101  return (header != *run_speedup_iter);
102  }))
103  {
104  //add speedup header in front of previous ordered speedup header if not first
105  //ordered header
106  if (run_speedup_iter != run_speedups_ordered.cbegin()) {
107  //find position in evaluation speedups of previous ordered header
108  //and add new header in front of it
109  speedup_headers_eval.insert(
110  (std::find(speedup_headers_eval.cbegin(), speedup_headers_eval.cend(), *(run_speedup_iter-1)) + 1),
111  *run_speedup_iter);
112  }
113  else {
114  //add speedup header to front of evaluation speedup headers if no previous
115  //ordered header in speedups for run
116  speedup_headers_eval.insert(speedup_headers_eval.cbegin(), *run_speedup_iter);
117  }
118  }
119  }
120  }
121 
122  //generate results across architectures
123  std::ostringstream result_across_archs_sstream;
124  //add text to display on top of results across architecture comparison file
125  for (const auto& comp_file_top_text_line : eval_across_runs_top_text) {
126  result_across_archs_sstream << comp_file_top_text_line << std::endl;
127  }
128  result_across_archs_sstream << std::endl;
129 
130  //write out the name of each input parameter to be displayed
131  for (const auto& input_param_disp_header : eval_across_runs_in_params_show) {
132  result_across_archs_sstream << input_param_disp_header << ',';
133  }
134 
135  //get header to use for speedup ordering
136  //use first speedup in speedup headers for ordering
137  //of runs from fastest to slowest
138  const auto speedup_ordering = speedup_headers.front();
139 
140  //write each evaluation run name and save order of runs with speedup corresponding to first
141  //speedup header
142  //order of run names is in speedup from largest to smallest
143  auto cmp_speedup =
144  [](const std::pair<std::string, float>& a, const std::pair<std::string, float>& b) {
145  return a.second > b.second; };
146  std::set<std::pair<std::string, float>, decltype(cmp_speedup)> run_names_in_order_w_speedup;
147 
148  //add first speedup with run name to order runs from fastest to slowest based on first speedup
149  for (const auto& [run_name, speedup_results] : speedup_results_name_to_data) {
150  if (speedup_results.contains(speedup_ordering)) {
151  const float avgSpeedupVsBase =
152  std::stof(std::string(speedup_results.at(speedup_ordering).at(0)));
153  run_names_in_order_w_speedup.insert({run_name, avgSpeedupVsBase});
154  }
155  else {
156  run_names_in_order_w_speedup.insert({run_name, 0});
157  }
158  }
159 
160  //write all the run names in order from fastest to slowest
161  for (const auto& [run_name, run_speedup] : run_names_in_order_w_speedup) {
162  result_across_archs_sstream << run_name << ',';
163  }
164  result_across_archs_sstream << std::endl;
165 
166  //write evaluation stereo set info, bp parameters, and total runtime for optimized bp implementation
167  //across each run in the evaluation
168  for (const auto& [input_sig, params_display_ordered] : inputs_to_params_disp_ordered) {
169  for (const auto& param_val_disp : params_display_ordered) {
170  result_across_archs_sstream << param_val_disp << ',';
171  }
172  for (const auto& [run_name, run_speedup] : run_names_in_order_w_speedup)
173  {
174  if (input_to_runtime_across_archs.at(run_name).contains(input_sig))
175  {
176  result_across_archs_sstream <<
177  input_to_runtime_across_archs.at(run_name).at(input_sig);
178  }
179  result_across_archs_sstream << ',';
180  }
181  result_across_archs_sstream << std::endl;
182  }
183  result_across_archs_sstream << std::endl;
184 
185  //write average speedup results for each run that correspond to a number of
186  //different evaluations of runtimes compared to a baseline
187  result_across_archs_sstream << "Average Speedups" << std::endl;
188  for (const auto& speedup_header : speedup_headers_eval) {
189  //don't process if header is empty
190  if (!(speedup_header.empty())) {
191  result_across_archs_sstream << speedup_header << ',';
192  //add empty cell for each input parameter after the first that's displayed
193  //so speedup shown in the same column same line as runtime
194  for (size_t i = 1; i < eval_across_runs_in_params_show.size(); i++) {
195  result_across_archs_sstream << ',';
196  }
197  //write speedup for each run in separate cells in horizontal direction
198  //where each column corresponds to a different evaluation run
199  for (const auto& [run_name, run_speedup] : run_names_in_order_w_speedup) {
200  if (speedup_results_name_to_data.at(run_name).contains(speedup_header)) {
201  result_across_archs_sstream <<
202  speedup_results_name_to_data.at(run_name).at(speedup_header).at(0) << ',';
203  }
204  else {
205  result_across_archs_sstream << ',';
206  }
207  }
208  //continue to next row of table to write data for next speedup result
209  result_across_archs_sstream << std::endl;
210  }
211  }
212 
213  //get file path for evaluation across runs and save evaluation across runs to csv file
214  const std::filesystem::path results_across_run_fp = imp_results_file_path /
215  (std::string(run_eval::kEvalAcrossRunsFileName) + std::string(run_eval::kCsvFileExtension));
216  std::ofstream eval_results_across_run_str(results_across_run_fp);
217  eval_results_across_run_str << result_across_archs_sstream.str();
218  std::cout << "Evaluation of results across all runs in " << results_across_run_fp << std::endl;
219 }

221 //function to get names of runs with results from implementation results file path
222 std::vector<std::string> EvaluateAcrossRuns::GetRunNames(
223  const std::filesystem::path& imp_results_file_path) const
224 {
225  //iterate through all run results files all run names with results and speedups
226  //create directory iterator with all results files
227  std::filesystem::directory_iterator results_files_iter =
228  std::filesystem::directory_iterator(imp_results_file_path / run_eval::kImpResultsRunDataFolderName);
229  std::filesystem::directory_iterator speedups_files_iter =
230  std::filesystem::directory_iterator(imp_results_file_path / run_eval::kImpResultsSpeedupsFolderName);
231  std::vector<std::string> run_names;
232  for (const auto& results_fp : results_files_iter)
233  {
234  std::string file_name_no_ext = results_fp.path().stem();
235  if (file_name_no_ext.ends_with("_" + std::string(run_eval::kRunResultsDescFileName)))
236  {
237  const std::string run_name =
238  file_name_no_ext.substr(
239  0, file_name_no_ext.find("_" + std::string(run_eval::kRunResultsDescFileName)));
240  run_names.push_back(run_name);
241  }
242  }
243 
244  //remove run name from runs to evaluate across runs if no speedup data
245  //example where this could be the case is baseline data that is used for comparison with other runs
246  for (auto run_names_iter = run_names.cbegin(); run_names_iter != run_names.cend();)
247  {
248  const std::filesystem::path run_speedup_fp =
249  imp_results_file_path / run_eval::kImpResultsSpeedupsFolderName /
250  (std::string(*run_names_iter) + '_' + std::string(run_eval::kSpeedupsDescFileName) +
251  std::string(run_eval::kCsvFileExtension));
252  //remove run from evaluation if no valid speedup data file that corresponds to run results data file
253  if ((!(std::filesystem::exists(run_speedup_fp))) ||
254  (!(std::filesystem::is_regular_file(run_speedup_fp))))
255  {
256  run_names_iter = run_names.erase(std::ranges::find(run_names, *run_names_iter));
257  }
258  else {
259  run_names_iter++;
260  }
261  }
262 
263  return run_names;
264 }
