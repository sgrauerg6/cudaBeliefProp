21 //create a Gaussian filter from a sigma value
22 std::pair<std::vector<float>, unsigned int> SmoothImage::MakeFilter(float sigma) const
23 {
24  const float sigma_use{std::max(sigma, 0.01f)};
25  const unsigned int size_filter{(unsigned int)std::ceil(sigma_use * kWidthSigma1) + 1u};
26  std::vector<float> mask(size_filter);
27  for (unsigned int i = 0; i < size_filter; i++) {
28  mask[i] = std::exp(-0.5*((i/sigma_use) * (i/sigma_use)));
29  }
30  NormalizeFilter(mask, size_filter);
31 
32  return {std::move(mask), size_filter};
33 }

35 //normalize filter mask so it integrates to one
36 void SmoothImage::NormalizeFilter(std::vector<float>& filter, unsigned int size_filter) const
37 {
38  float sum{0.0};
39  for (unsigned int i = 1; i < size_filter; i++) {
40  sum += std::abs(filter[i]);
41  }
42  sum = 2*sum + std::abs(filter[0]);
43  for (unsigned int i = 0; i < size_filter; i++) {
44  filter[i] /= sum;
45  }
46 }
