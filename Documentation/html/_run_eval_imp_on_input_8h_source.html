56  //run optimized and single threaded implementations using multiple sets of
57  //parallel parameters in optimized implementation if set to optimize parallel
58  //parameters
59  //returns data from runs using default and optimized parallel parameters
60  MultRunData::mapped_type RunEvalBenchmark(
61  const run_environment::RunImpSettings& run_imp_settings,
62  bool run_w_loop_iters_templated) const
63  {
64  MultRunData::mapped_type::value_type out_run_data;
65  enum class RunType { ONLY_RUN, DEFAULT_PARAMS, OPTIMIZED_RUN, TEST_PARAMS };
66 
67  //set up parallel parameters for specific benchmark
68  std::shared_ptr<ParallelParams> parallel_params = SetUpParallelParams(run_imp_settings);
69 
70  //if optimizing parallel parameters, parallel_params_vect contains parallel parameter settings to run
71  //(and contains only the default parallel parameters if not)
72  std::vector<std::array<unsigned int, 2>> parallel_params_vect{
73  run_imp_settings.opt_parallel_params_setting.first ?
74  run_imp_settings.p_params_default_opt_settings.second :
75  std::vector<std::array<unsigned int, 2>>()};
76 
77  //if optimizing parallel parameters, run BP for each parallel parameters option,
78  //retrieve best parameters for each kernel or overall for the run,
79  //and then run BP with best found parallel parameters
80  //if not optimizing parallel parameters, run BP once using default parallel parameters
81  for (unsigned int run_num = 0; run_num < (parallel_params_vect.size() + 1); run_num++) {
82  //initialize current run type to specify if current run is only run, run with
83  //default params, test params run, or final run with optimized params
84  RunType curr_run_type{RunType::TEST_PARAMS};
85  if (!run_imp_settings.opt_parallel_params_setting.first) {
86  curr_run_type = RunType::ONLY_RUN;
87  }
88  else if (run_num == parallel_params_vect.size()) {
89  curr_run_type = RunType::OPTIMIZED_RUN;
90  }
91 
92  //get and set parallel parameters for current run if not final run that uses optimized parameters
93  std::array<unsigned int, 2> p_params_curr_run{run_imp_settings.p_params_default_opt_settings.first};
94  if (curr_run_type == RunType::ONLY_RUN) {
95  parallel_params->SetParallelDims(run_imp_settings.p_params_default_opt_settings.first);
96  }
97  else if (curr_run_type == RunType::TEST_PARAMS) {
98  //set parallel parameters to parameters corresponding to current run for each BP processing level
99  p_params_curr_run = parallel_params_vect[run_num];
100  parallel_params->SetParallelDims(p_params_curr_run);
101  if (p_params_curr_run == run_imp_settings.p_params_default_opt_settings.first) {
102  //set run type to default parameters if current run uses default parameters
103  curr_run_type = RunType::DEFAULT_PARAMS;
104  }
105  }
106 
107  //store input params data if using default parallel parameters or final run with optimized parameters
108  RunData curr_run_data;
109  if (curr_run_type != RunType::TEST_PARAMS) {
110  //add input and parameters data for specific benchmark to current run data
111  curr_run_data.AddDataWHeader(
112  std::string(run_eval::kInputIdxHeader),
113  std::to_string(NUM_INPUT));
114  curr_run_data.AppendData(InputAndParamsForCurrBenchmark(run_w_loop_iters_templated));
115  if ((run_imp_settings.opt_parallel_params_setting.first) &&
116  (run_imp_settings.opt_parallel_params_setting.second ==
117  run_environment::OptParallelParamsSetting::kAllowDiffKernelParallelParamsInRun))
118  {
119  //add parallel parameters for each kernel to current input data if
120  //allowing different parallel parameters for each kernel
121  //in the same run
122  curr_run_data.AppendData(parallel_params->AsRunData());
123  }
124  }
125 
126  //run only optimized implementation and not single-threaded run if
127  //current run is not final run or is using default parameter parameters
128  const bool run_opt_imp_only{curr_run_type == RunType::TEST_PARAMS};
129 
130  //run benchmark implementation(s) and return null output if error in run
131  //detailed results stored to file that is generated using stream
132  const auto run_imps_results =
133  RunImpsAndCompare(parallel_params, run_opt_imp_only, run_w_loop_iters_templated);
134  curr_run_data.AddDataWHeader(
135  std::string(run_eval::kRunSuccessHeader),
136  run_imps_results.has_value());
137 
138  //if error in run and run is any type other than for testing parameters,
139  //exit function with null output to indicate error
140  if ((!run_imps_results) && (curr_run_type != RunType::TEST_PARAMS)) {
141  return {};
142  }
143 
144  //add data results from current run if run successful
145  if (run_imps_results) {
146  curr_run_data.AppendData(run_imps_results.value());
147  }
148 
149  //add current run results for output if using default parallel
150  //parameters or is final run w/ optimized parallel parameters
151  if (curr_run_type != RunType::TEST_PARAMS) {
152  //set output for runs using default parallel parameters and final run
153  //(which is the same run if not optimizing parallel parameters)
154  if (curr_run_type == RunType::OPTIMIZED_RUN) {
155  out_run_data[run_environment::ParallelParamsSetting::kOptimized] = curr_run_data;
156  }
157  else {
158  out_run_data[run_environment::ParallelParamsSetting::kDefault] = curr_run_data;
159  if (curr_run_type == RunType::ONLY_RUN) {
160  //if current run is only run due to setting of not optimizing parallel parameters,
161  //set optimized parallel parameters output to current run data, resulting in same
162  //run data mapped to default and optimized parallel parameters enums
163  out_run_data[run_environment::ParallelParamsSetting::kOptimized] = curr_run_data;
164  }
165  }
166  }
167 
168  if (run_imp_settings.opt_parallel_params_setting.first) {
169  //retrieve and store results including runtimes for each kernel if
170  //allowing different parallel parameters for each kernel and
171  //total runtime for current run
172  //if error in run, don't add results for current parallel parameters to results set
173  if (run_imps_results) {
174  if (curr_run_type != RunType::OPTIMIZED_RUN) {
175  parallel_params->AddTestResultsForParallelParams(p_params_curr_run, curr_run_data);
176  }
177  }
178 
179  //set optimized parallel parameters if next run is final run
180  //that uses optimized parallel parameters
181  //optimized parallel parameters are determined from previous test runs
182  //using multiple test parallel parameters
183  if (run_num == (parallel_params_vect.size() - 1)) {
184  parallel_params->SetOptimizedParams();
185  }
186  }
187  }
188 
189  return out_run_data;
190  }
191 };
